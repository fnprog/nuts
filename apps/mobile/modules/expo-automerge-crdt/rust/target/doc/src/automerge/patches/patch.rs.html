11/// A change to the current state of the document
12///
13/// [`Patch`]es are obtained from a [`PatchLog`](super::PatchLog) which has been passed to any of
14/// the various methods which mutate a document and add incremental changes to the
15/// [`PatchLog`](super::PatchLog)
16#[derive(Debug, Clone, PartialEq)]
17pub struct Patch {
18    /// The object this patch modifies
19    pub obj: ObjId,
20    /// The path to the property in the parent object where this object lives
21    pub path: Vec<(ObjId, Prop)>,
22    /// The change this patch represents
23    pub action: PatchAction,
24}
25
26#[derive(Debug, Clone, PartialEq)]
27pub enum PatchAction {
28    /// A key was created or updated in a map
29    PutMap {
30        key: String,
31        /// The value that was inserted and the object ID of the object that was inserted. Note
32        /// that the Object ID is only meaningful for `Value::Obj` values
33        value: (Value<'static>, ObjId),
34        /// Whether there is a conflict at this key. If there is a conflict this patch represents
35        /// the "winning" value of the conflict. The conflicting values can be obtained with
36        /// [`crate::ReadDoc::get_all`]
37        conflict: bool,
38    },
39    /// An index in a sequence was updated
40    PutSeq {
41        index: usize,
42        /// The value that was set and the object ID of the object that was set. Note that the
43        /// Object ID is only meaningful for `Value::Obj` values
44        value: (Value<'static>, ObjId),
45        /// Whether there is a conflict at this index. If there is a conflict this patch represents
46        /// the "winning" value of the conflict. The conflicting values can be obtained with
47        /// [`crate::ReadDoc::get_all`]
48        conflict: bool,
49    },
50    /// One or more elements were inserted into a sequence
51    Insert {
52        index: usize,
53        /// The values that were inserted, in order that they appear. As with [`Self::PutMap`] and
54        /// [`Self::PutSeq`] the object ID is only meaningful for `Value::Obj` values
55        values: SequenceTree<(Value<'static>, ObjId, bool)>,
56    },
57    /// Some text was spliced into a text object
58    SpliceText {
59        index: usize,
60        /// The text that was inserted
61        value: TextValue,
62        /// All marks currently active for this span of text
63        marks: Option<MarkSet>,
64    },
65    /// A counter was incremented
66    Increment {
67        /// The property of the counter that was incremented within the target object
68        prop: Prop,
69        /// The amount incremented, may be negative
70        value: i64,
71    },
72    /// A new conflict has appeared
73    Conflict {
74        /// The conflicted property
75        prop: Prop,
76    },
77    /// A key was deleted from a map
78    DeleteMap { key: String },
79    /// One or more indices were removed from a sequence
80    DeleteSeq { index: usize, length: usize },
81    /// Some marks within a text object were added or removed
82    Mark { marks: Vec<Mark<'static>> },
83}
84
85impl fmt::Display for PatchAction {
86    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
87        write!(f, "{:?}", self)
88    }
89}
