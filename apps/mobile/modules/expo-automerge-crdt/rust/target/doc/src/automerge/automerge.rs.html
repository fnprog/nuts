103    /// Whether to convert [`ScalarValue::Str`]s in the loaded document to [`ObjType::Text`]
104    ///
105    /// Until version 2.1.0 of the javascript library strings (as in, the native string of the JS
106    /// runtime) were represented in the document as [`ScalarValue::Str`] and there was a special
107    /// JS class called `Text` which users were expected to use for [`ObjType::Text`]. In `2.1.0`
108    /// we changed this so that native strings were represented as [`ObjType::Text`] and
109    /// [`ScalarValue::Str`] was represented as a special `RawString` class. This means
110    /// that upgrading the application code to use the new API would require either
111    ///
112    /// a) Maintaining two code paths in the application to deal with both `string` and `RawString`
113    ///    types
114    /// b) Writing a migration script to convert all `RawString` types to `string`
115    ///
116    /// The latter is logic which is the same for all applications so we implement it in the
117    /// library for convenience. The way this works is that after loading the document we iterate
118    /// through all visible [`ScalarValue::Str`] values and emit a change which creates a new
119    /// [`ObjType::Text`] at the same path with the same content.
120    pub fn migrate_strings(self, migration: StringMigration) -> Self {
121        Self {
122            string_migration: migration,
123            ..self
124        }
125    }

139/// An automerge document which does not manage transactions for you.
140///
141/// ## Creating, loading, merging and forking documents
142///
143/// A new document can be created with [`Self::new()`], which will create a document with a random
144/// [`ActorId`]. Existing documents can be loaded with [`Self::load()`], or [`Self::load_with()`].
145///
146/// If you have two documents and you want to merge the changes from one into the other you can use
147/// [`Self::merge()`] or [`Self::merge_and_log_patches()`].
148///
149/// If you have a document you want to split into two concurrent threads of execution you can use
150/// [`Self::fork()`]. If you want to split a document from ealier in its history you can use
151/// [`Self::fork_at()`].
152///
153/// ## Reading values
154///
155/// [`Self`] implements [`ReadDoc`], which provides methods for reading values from the document.
156///
157/// ## Modifying a document (Transactions)
158///
159/// [`Automerge`] provides an interface for viewing and modifying automerge documents which does
160/// not manage transactions for you. To create changes you use either [`Automerge::transaction()`] or
161/// [`Automerge::transact()`] (or the `_with` variants).
162///
163/// ## Sync
164///
165/// This type implements [`crate::sync::SyncDoc`]
166///
167#[derive(Debug, Clone)]
168pub struct Automerge {
169    /// The list of unapplied changes that are not causally ready.
170    queue: Vec<Change>,
171    /// The history of changes that form this document, topologically sorted too.
172    history: Vec<Change>,
173    /// Mapping from change hash to index into the history list.
174    history_index: HashMap<ChangeHash, usize>,
175    /// Graph of changes
176    change_graph: ChangeGraph,
177    /// Mapping from actor index to list of seqs seen for them.
178    states: HashMap<usize, Vec<usize>>,
179    /// Current dependencies of this document (heads hashes).
180    deps: HashSet<ChangeHash>,
181    /// The set of operations that form this document.
182    ops: OpSet,
183    /// The current actor.
184    actor: Actor,
185    /// The maximum operation counter this document has seen.
186    max_op: u64,
187}

286    pub(crate) fn transaction_args(&mut self, heads: Option<&[ChangeHash]>) -> TransactionArgs {
287        let actor_index;
288        let seq;
289        let mut deps;
290        let scope;
291        match heads {
292            Some(heads) => {
293                deps = heads.to_vec();
294                let isolation = self.isolate_actor(heads);
295                actor_index = isolation.actor_index;
296                seq = isolation.seq;
297                scope = Some(isolation.clock);
298            }
299            None => {
300                actor_index = self.get_actor_index();
301                seq = self.states.get(&actor_index).map_or(0, |v| v.len()) as u64 + 1;
302                deps = self.get_heads();
303                scope = None;
304                if seq > 1 {
305                    let last_hash = self.get_hash(actor_index, seq - 1).unwrap();
306                    if !deps.contains(&last_hash) {
307                        deps.push(last_hash);
308                    }
309                }
310            }
311        }
312
313        // SAFETY: this unwrap is safe as we always add 1
314        let start_op = NonZeroU64::new(self.max_op + 1).unwrap();
315        let idx_range = self.osd().start_range();
316        TransactionArgs {
317            actor_index,
318            seq,
319            start_op,
320            idx_range,
321            deps,
322            scope,
323        }
324    }

453    /// Fork this document at the given heads
454    ///
455    /// This will create a new actor ID for the forked document
456    pub fn fork_at(&self, heads: &[ChangeHash]) -> Result<Self, AutomergeError> {
457        let mut seen = heads.iter().cloned().collect::<HashSet<_>>();
458        let mut heads = heads.to_vec();
459        let mut changes = vec![];
460        while let Some(hash) = heads.pop() {
461            if let Some(idx) = self.history_index.get(&hash) {
462                let change = &self.history[*idx];
463                for dep in change.deps() {
464                    if !seen.contains(dep) {
465                        heads.push(*dep);
466                    }
467                }
468                changes.push(change);
469                seen.insert(hash);
470            } else {
471                return Err(AutomergeError::InvalidHash(hash));
472            }
473        }
474        let mut f = Self::new();
475        f.set_actor(ActorId::random());
476        f.apply_changes(changes.into_iter().rev().cloned())?;
477        Ok(f)
478    }

578    /// Load a document, with options
579    ///
580    /// # Arguments
581    /// * `data` - The data to load
582    /// * `options` - The options to use when loading
583    #[tracing::instrument(skip(data), err)]
584    pub fn load_with_options<'a, 'b>(
585        data: &'a [u8],
586        options: LoadOptions<'b>,
587    ) -> Result<Self, AutomergeError> {
588        if data.is_empty() {
589            tracing::trace!("no data, initializing empty document");
590            return Ok(Self::new());
591        }
592        tracing::trace!("loading first chunk");
593        let (remaining, first_chunk) = storage::Chunk::parse(storage::parse::Input::new(data))
594            .map_err(|e| load::Error::Parse(Box::new(e)))?;
595        if !first_chunk.checksum_valid() {
596            return Err(load::Error::BadChecksum.into());
597        }
598
599        let mut change: Option<Change> = None;
600        let mut first_chunk_was_doc = false;
601        let mut am = match first_chunk {
602            storage::Chunk::Document(d) => {
603                tracing::trace!("first chunk is document chunk, inflating");
604                first_chunk_was_doc = true;
605                reconstruct_document(&d, options.verification_mode)?
606            }
607            storage::Chunk::Change(stored_change) => {
608                tracing::trace!("first chunk is change chunk");
609                change = Some(
610                    Change::new_from_unverified(stored_change.into_owned(), None)
611                        .map_err(|e| load::Error::InvalidChangeColumns(Box::new(e)))?,
612                );
613                Self::new()
614            }
615            storage::Chunk::CompressedChange(stored_change, compressed) => {
616                tracing::trace!("first chunk is compressed change");
617                change = Some(
618                    Change::new_from_unverified(
619                        stored_change.into_owned(),
620                        Some(compressed.into_owned()),
621                    )
622                    .map_err(|e| load::Error::InvalidChangeColumns(Box::new(e)))?,
623                );
624                Self::new()
625            }
626        };
627        tracing::trace!("loading change chunks");
628        match load::load_changes(remaining.reset()) {
629            load::LoadedChanges::Complete(c) => {
630                am.apply_changes(change.into_iter().chain(c))?;
631                // Only allow missing deps if the first chunk was a document chunk
632                // See https://github.com/automerge/automerge/pull/599#issuecomment-1549667472
633                if !am.queue.is_empty()
634                    && !first_chunk_was_doc
635                    && options.on_partial_load == OnPartialLoad::Error
636                {
637                    return Err(AutomergeError::MissingDeps);
638                }
639            }
640            load::LoadedChanges::Partial { error, .. } => {
641                if options.on_partial_load == OnPartialLoad::Error {
642                    return Err(error.into());
643                }
644            }
645        }
646        if let StringMigration::ConvertToText = options.string_migration {
647            am.convert_scalar_strings_to_text()?;
648        }
649        if let Some(patch_log) = options.patch_log {
650            if patch_log.is_active() {
651                current_state::log_current_state_patches(&am, patch_log);
652            }
653        }
654        Ok(am)
655    }

689    /// Like [`Self::load_incremental()`] but log the changes to the current state of the document
690    /// to [`PatchLog`]
691    pub fn load_incremental_log_patches(
692        &mut self,
693        data: &[u8],
694        patch_log: &mut PatchLog,
695    ) -> Result<usize, AutomergeError> {
696        if self.is_empty() {
697            let mut doc = Self::load_with_options(
698                data,
699                LoadOptions::new()
700                    .on_partial_load(OnPartialLoad::Ignore)
701                    .verification_mode(VerificationMode::Check),
702            )?;
703            doc = doc.with_actor(self.actor_id());
704            if patch_log.is_active() {
705                current_state::log_current_state_patches(&doc, patch_log);
706            }
707            *self = doc;
708            return Ok(self.ops.len());
709        }
710        let changes = match load::load_changes(storage::parse::Input::new(data)) {
711            load::LoadedChanges::Complete(c) => c,
712            load::LoadedChanges::Partial { error, loaded, .. } => {
713                tracing::warn!(successful_chunks=loaded.len(), err=?error, "partial load");
714                loaded
715            }
716        };
717        let start = self.ops.len();
718        self.apply_changes_log_patches(changes, patch_log)?;
719        let delta = self.ops.len() - start;
720        Ok(delta)
721    }

739        changes: impl IntoIterator<Item = Change>,

747    /// Like [`Self::apply_changes()`] but log the resulting changes to the current state of the
748    /// document to `patch_log`
749    pub fn apply_changes_log_patches<I: IntoIterator<Item = Change>>(
750        &mut self,
751        changes: I,
752        patch_log: &mut PatchLog,
753    ) -> Result<(), AutomergeError> {
754        // Record this so we can avoid observing each individual change and instead just observe
755        // the final state after all the changes have been applied. We can only do this for an
756        // empty document right now, once we have logic to produce the diffs between arbitrary
757        // states of the OpSet we can make this cleaner.
758        for c in changes {
759            if !self.history_index.contains_key(&c.hash()) {
760                if self.duplicate_seq(&c) {
761                    return Err(AutomergeError::DuplicateSeqNumber(
762                        c.seq(),
763                        c.actor_id().clone(),
764                    ));
765                }
766                if self.is_causally_ready(&c) {
767                    self.apply_change(c, patch_log)?;
768                } else {
769                    self.queue.push(c);
770                }
771            }
772        }
773        while let Some(c) = self.pop_next_causally_ready_change() {
774            if !self.history_index.contains_key(&c.hash()) {
775                self.apply_change(c, patch_log)?;
776            }
777        }
778        Ok(())
779    }

781    fn apply_change(
782        &mut self,
783        change: Change,
784        patch_log: &mut PatchLog,
785    ) -> Result<(), AutomergeError> {
786        let ops = self.import_ops(&change);
787        self.update_history(change, ops.len());
788        for (obj, op, pred) in ops {
789            self.insert_op(&obj, op, &pred, patch_log)?;
790        }
791        Ok(())
792    }

801    fn pop_next_causally_ready_change(&mut self) -> Option<Change> {
802        let mut index = 0;
803        while index < self.queue.len() {
804            if self.is_causally_ready(&self.queue[index]) {
805                return Some(self.queue.swap_remove(index));
806            }
807            index += 1;
808        }
809        None
810    }

812    fn import_ops(&mut self, change: &Change) -> Vec<(ObjId, OpBuilder, OpIds)> {
813        let actor = self.ops.osd.actors.cache(change.actor_id().clone());
814        let mut actors = Vec::with_capacity(change.other_actor_ids().len() + 1);
815        actors.push(actor);
816        actors.extend(
817            change
818                .other_actor_ids()
819                .iter()
820                .map(|a| self.ops.osd.actors.cache(a.clone()))
821                .collect::<Vec<_>>(),
822        );
823        change
824            .iter_ops()
825            .enumerate()
826            .map(|(i, c)| {
827                let id = OpId::new(change.start_op().get() + i as u64, actor);
828                let key = match &c.key {
829                    EncodedKey::Prop(n) => Key::Map(self.ops.osd.props.cache(n.to_string())),
830                    EncodedKey::Elem(e) if e.is_head() => Key::Seq(ElemId::head()),
831                    EncodedKey::Elem(ElemId(o)) => {
832                        Key::Seq(ElemId(OpId::new(o.counter(), actors[o.actor()])))
833                    }
834                };
835                let obj = if c.obj.is_root() {
836                    ObjId::root()
837                } else {
838                    ObjId(OpId::new(
839                        c.obj.opid().counter(),
840                        actors[c.obj.opid().actor()],
841                    ))
842                };
843                let pred = c
844                    .pred
845                    .iter()
846                    .map(|p| OpId::new(p.counter(), actors[p.actor()]));
847                let pred = self.ops.osd.sorted_opids(pred);
848                (
849                    obj,
850                    OpBuilder {
851                        id,
852                        action: OpType::from_action_and_value(
853                            c.action,
854                            c.val,
855                            c.mark_name,
856                            c.expand,
857                        ),
858                        key,
859                        insert: c.insert,
860                    },
861                    pred,
862                )
863            })
864            .collect()
865    }

972    /// Get the changes since `have_deps` in this document using a clock internally.
973    fn get_changes_clock(&self, have_deps: &[ChangeHash]) -> Vec<&Change> {
974        // get the clock for the given deps
975        let clock = self.clock_at(have_deps);
976
977        // get the documents current clock
978
979        let mut change_indexes: Vec<usize> = Vec::new();
980        // walk the state from the given deps clock and add them into the vec
981        for (actor_index, actor_changes) in &self.states {
982            if let Some(clock_data) = clock.get_for_actor(actor_index) {
983                // find the change in this actors sequence of changes that corresponds to the max_op
984                // recorded for them in the clock
985                change_indexes.extend(&actor_changes[clock_data.seq as usize..]);
986            } else {
987                change_indexes.extend(&actor_changes[..]);
988            }
989        }
990
991        // ensure the changes are still in sorted order
992        change_indexes.sort_unstable();
993
994        change_indexes
995            .into_iter()
996            .map(|i| &self.history[i])
997            .collect()
998    }

1023    pub(crate) fn isolate_actor(&mut self, heads: &[ChangeHash]) -> Isolation {
1024        let mut clock = self.clock_at(heads);
1025        let mut actor_index = self.get_isolated_actor_index(0);
1026
1027        for i in 1.. {
1028            let max_op = self.max_op_for_actor(actor_index);
1029            if max_op == 0 || clock.covers(&OpId::new(max_op, actor_index)) {
1030                clock.isolate(actor_index);
1031                break;
1032            }
1033            actor_index = self.get_isolated_actor_index(i);
1034        }
1035
1036        let seq = self.states.get(&actor_index).map_or(0, |v| v.len()) as u64 + 1;
1037
1038        Isolation {
1039            actor_index,
1040            seq,
1041            clock,
1042        }
1043    }

1063    pub(crate) fn update_history(&mut self, change: Change, num_ops: usize) -> usize {
1064        self.max_op = std::cmp::max(self.max_op, change.start_op().get() + num_ops as u64 - 1);
1065
1066        self.update_deps(&change);
1067
1068        let history_index = self.history.len();
1069
1070        let actor_index = self.ops.osd.actors.cache(change.actor_id().clone());
1071        self.states
1072            .entry(actor_index)
1073            .or_default()
1074            .push(history_index);
1075
1076        self.history_index.insert(change.hash(), history_index);
1077        self.change_graph
1078            .add_change(&change, actor_index)
1079            .expect("Change's deps should already be in the document");
1080
1081        self.history.push(change);
1082
1083        history_index
1084    }

1206        &mut self,
1207        obj: &ObjId,
1208        op: OpBuilder,
1209        pred: &OpIds,
1210        patch_log: &mut PatchLog,
1211    ) -> Result<(), AutomergeError> {
1212        let is_delete = op.is_delete();
1213        let idx = self.ops.load(*obj, op);
1214        let op = idx.as_op(&self.ops.osd);
1215
1216        let (pos, succ) = if patch_log.is_active() {
1217            let obj = self.get_obj_meta(*obj)?;
1218            let found = self.ops.find_op_with_patch_log(
1219                &obj,
1220                patch_log.text_rep().encoding(obj.typ),
1221                op,
1222                pred,
1223            );
1224            found.log_patches(&obj, op, pred, self, patch_log);
1225            (found.pos, found.succ)
1226        } else {
1227            let found = self.ops.find_op_without_patch_log(obj, op, pred);
1228            (found.pos, found.succ)
1229        };
1230
1231        self.ops.add_succ(obj, &succ, idx);
1232
1233        if !is_delete {
1234            self.ops.insert(pos, obj, idx);
1235        }
1236        Ok(())
1237    }

1267    /// Get changes in `other` that are not in `self`
1268    pub fn get_changes_added<'a>(&self, other: &'a Self) -> Vec<&'a Change> {
1269        // Depth-first traversal from the heads through the dependency graph,
1270        // until we reach a change that is already present in other
1271        let mut stack: Vec<_> = other.get_heads();
1272        tracing::trace!(their_heads=?stack, "finding changes to merge");
1273        let mut seen_hashes = HashSet::new();
1274        let mut added_change_hashes = Vec::new();
1275        while let Some(hash) = stack.pop() {
1276            if !seen_hashes.contains(&hash) && self.get_change_by_hash(&hash).is_none() {
1277                seen_hashes.insert(hash);
1278                added_change_hashes.push(hash);
1279                if let Some(change) = other.get_change_by_hash(&hash) {
1280                    stack.extend(change.deps());
1281                }
1282            }
1283        }
1284        // Return those changes in the reverse of the order in which the depth-first search
1285        // found them. This is not necessarily a topological sort, but should usually be close.
1286        added_change_hashes.reverse();
1287        added_change_hashes
1288            .into_iter()
1289            .filter_map(|h| other.get_change_by_hash(&h))
1290            .collect()
1291    }

1300            ExId::Root => None,
1301            ExId::Id(..) => {
1302                let opid = self.exid_to_opid(exid).ok()?;
1303                let actor_indices = self.states.get(&opid.actor())?;
1304                let change_index_index = actor_indices
1305                    .binary_search_by(|change_index| {
1306                        let change = self
1307                            .history
1308                            .get(*change_index)
1309                            .expect("State index should refer to a valid change");
1310                        let start = change.start_op().get();
1311                        let len = change.len() as u64;
1312                        if opid.counter() < start {
1313                            Ordering::Greater
1314                        } else if start + len <= opid.counter() {
1315                            Ordering::Less
1316                        } else {
1317                            Ordering::Equal
1318                        }
1319                    })
1320                    .ok()?;
1321                let change_index = actor_indices.get(change_index_index).unwrap();
1322                Some(self.history.get(*change_index).unwrap().hash())
1323            }
1324        }
1325    }

1327    fn calculate_marks(
1328        &self,
1329        obj: &ExId,
1330        clock: Option<Clock>,
1331    ) -> Result<Vec<Mark<'_>>, AutomergeError> {
1332        let obj = self.exid_to_obj(obj.as_ref())?;
1333        let ops_by_key = self.ops().iter_ops(&obj.id).chunk_by(|o| o.elemid_or_key());
1334        let mut index = 0;
1335        let mut marks = MarkStateMachine::default();
1336        let mut acc = MarkAccumulator::default();
1337        let mut last_marks = None;
1338        let mut mark_len = 0;
1339        let mut mark_index = 0;
1340        for (_key, key_ops) in ops_by_key.into_iter() {
1341            if let Some(o) = key_ops.filter(|o| o.visible_or_mark(clock.as_ref())).last() {
1342                match o.action() {
1343                    OpType::Make(_) | OpType::Put(_) => {
1344                        let len = o.width(TextRepresentation::String.encoding(obj.typ));
1345                        if last_marks.as_ref() != marks.current() {
1346                            match last_marks.as_ref() {
1347                                Some(m) if mark_len > 0 => acc.add(mark_index, mark_len, m),
1348                                _ => (),
1349                            }
1350                            last_marks = marks.current().cloned();
1351                            mark_index = index;
1352                            mark_len = 0;
1353                        }
1354                        mark_len += len;
1355                        index += len;
1356                    }
1357                    OpType::MarkBegin(_, data) => {
1358                        marks.mark_begin(*o.id(), data, &self.ops.osd);
1359                    }
1360                    OpType::MarkEnd(_) => {
1361                        marks.mark_end(*o.id(), &self.ops.osd);
1362                    }
1363                    OpType::Increment(_) | OpType::Delete => {}
1364                }
1365            }
1366        }
1367        match last_marks.as_ref() {
1368            Some(m) if mark_len > 0 => acc.add(mark_index, mark_len, m),
1369            _ => (),
1370        }
1371        Ok(acc.into_iter_no_unmark().collect())
1372    }

1606    fn convert_scalar_strings_to_text(&mut self) -> Result<(), AutomergeError> {
1607        struct Conversion {
1608            obj_id: ExId,
1609            prop: Prop,
1610            text: smol_str::SmolStr,
1611        }
1612        let mut to_convert = Vec::new();
1613        for (obj, ops) in self.ops.iter_objs() {
1614            match obj.typ {
1615                ObjType::Map | ObjType::List => {
1616                    for op in ops {
1617                        let op = op.as_op(self.osd());
1618                        if !op.visible() {
1619                            continue;
1620                        }
1621                        if let OpType::Put(ScalarValue::Str(s)) = op.action() {
1622                            let prop = match *op.key() {
1623                                Key::Map(prop) => Prop::Map(self.ops.osd.props.get(prop).clone()),
1624                                Key::Seq(_) => {
1625                                    let Some(found) = self.ops.seek_list_opid(
1626                                        &obj.id,
1627                                        *op.id(),
1628                                        ListEncoding::List,
1629                                        None,
1630                                    ) else {
1631                                        continue;
1632                                    };
1633                                    Prop::Seq(found.index)
1634                                }
1635                            };
1636                            to_convert.push(Conversion {
1637                                obj_id: self.ops.id_to_exid(obj.id.0),
1638                                prop,
1639                                text: s.clone(),
1640                            })
1641                        }
1642                    }
1643                }
1644                _ => {}
1645            }
1646        }
1647
1648        if !to_convert.is_empty() {
1649            let mut tx = self.transaction();
1650            for Conversion { obj_id, prop, text } in to_convert {
1651                let text_id = tx.put_object(obj_id, prop, ObjType::Text)?;
1652                tx.splice_text(&text_id, 0, 0, &text)?;
1653            }
1654            tx.commit();
1655        }
1656
1657        Ok(())
1658    }

1660    pub(crate) fn visible_obj_paths(
1661        &self,
1662        at: Option<&[ChangeHash]>,
1663    ) -> HashMap<ExId, Vec<(ExId, Prop)>> {
1664        let at = at.map(|heads| self.clock_at(heads));
1665        let mut paths = HashMap::<ExId, Vec<(ExId, Prop)>>::new();
1666        let mut visible_objs = HashSet::<crate::types::ObjId>::new();
1667        visible_objs.insert(crate::types::ObjId::root());
1668        paths.insert(ExId::Root, vec![]);
1669
1670        for (obj, ops) in self.ops.iter_objs() {
1671            // Note that this works because the OpSet iterates in causal order,
1672            // which means that we have already seen the operation which
1673            // creates the object and added it to the visible_objs set if it
1674            // is visible.
1675            if !visible_objs.contains(&obj.id) {
1676                continue;
1677            }
1678            for op_idx in ops {
1679                let op = op_idx.as_op(self.osd());
1680                if op.visible_at(at.as_ref()) {
1681                    if let OpType::Make(_) = op.action() {
1682                        visible_objs.insert(op.id().into());
1683                        let (mut path, parent_obj_id) = if obj.id.is_root() {
1684                            (vec![], ExId::Root)
1685                        } else {
1686                            let parent_obj_id = self.ops.id_to_exid(obj.id.into());
1687                            (paths.get(&parent_obj_id).cloned().unwrap(), parent_obj_id)
1688                        };
1689                        let prop = match op.key() {
1690                            Key::Map(prop) => Prop::Map(self.ops.osd.props.get(*prop).clone()),
1691                            Key::Seq(_) => {
1692                                let encoding = match obj.typ {
1693                                    ObjType::Text => ListEncoding::Text,
1694                                    _ => ListEncoding::List,
1695                                };
1696                                let found = self
1697                                    .ops
1698                                    .seek_list_opid(&obj.id, *op.id(), encoding, at.as_ref())
1699                                    .unwrap();
1700                                Prop::Seq(found.index)
1701                            }
1702                        };
1703                        path.push((parent_obj_id.clone(), prop));
1704                        let obj_id = self.ops.id_to_exid(*op.id());
1705                        paths.insert(obj_id, path);
1706                    }
1707                }
1708            }
1709        }
1710        paths
1711    }

1916        let obj = obj.as_ref();
1917        let opid = self.exid_to_opid(obj)?;
1918        let typ = self.ops.object_type(&ObjId(opid));
1919        typ.ok_or_else(|| AutomergeError::InvalidObjId(obj.to_string()))

1922    fn get_missing_deps(&self, heads: &[ChangeHash]) -> Vec<ChangeHash> {
1923        let in_queue: HashSet<_> = self.queue.iter().map(|change| change.hash()).collect();
1924        let mut missing = HashSet::new();
1925
1926        for head in self.queue.iter().flat_map(|change| change.deps()) {
1927            if !self.history_index.contains_key(head) {
1928                missing.insert(head);
1929            }
1930        }
1931
1932        for head in heads {
1933            if !self.history_index.contains_key(head) {
1934                missing.insert(head);
1935            }
1936        }
1937
1938        let mut missing = missing
1939            .into_iter()
1940            .filter(|hash| !in_queue.contains(hash))
1941            .copied()
1942            .collect::<Vec<_>>();
1943        missing.sort();
1944        missing
1945    }

1950            .and_then(|index| self.history.get(*index))
1951    }

1998pub(crate) fn reconstruct_document<'a>(
1999    doc: &'a storage::Document<'a>,
2000    mode: VerificationMode,
2001) -> Result<Automerge, AutomergeError> {
2002    let storage::load::ReconOpSet {
2003        changes,
2004        op_set,
2005        heads,
2006        max_op,
2007    } = storage::load::reconstruct_opset(doc, mode)
2008        .map_err(|e| load::Error::InflateDocument(Box::new(e)))?;
2009
2010    let mut hashes_by_index = HashMap::new();
2011    let mut actor_to_history: HashMap<usize, Vec<usize>> = HashMap::new();
2012    let mut change_graph = ChangeGraph::new();
2013    for (index, change) in changes.iter().enumerate() {
2014        // SAFETY: This should be fine because we just constructed an opset containing
2015        // all the changes
2016        let actor_index = op_set.osd.actors.lookup(change.actor_id()).unwrap();
2017        actor_to_history.entry(actor_index).or_default().push(index);
2018        hashes_by_index.insert(index, change.hash());
2019        change_graph.add_change(change, actor_index)?;
2020    }
2021    let history_index = hashes_by_index.into_iter().map(|(k, v)| (v, k)).collect();
2022    Ok(Automerge {
2023        queue: vec![],
2024        history: changes,
2025        history_index,
2026        states: actor_to_history,
2027        change_graph,
2028        ops: op_set,
2029        deps: heads.into_iter().collect(),
2030        actor: Actor::Unused(ActorId::random()),
2031        max_op,
2032    })
