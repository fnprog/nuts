13#[derive(Debug, thiserror::Error)]
14#[allow(unreachable_pub)]
15pub enum Error {
16    #[error("unable to parse chunk: {0}")]
17    Parse(Box<dyn std::error::Error + Send + Sync + 'static>),
18    #[error("invalid change columns: {0}")]
19    InvalidChangeColumns(Box<dyn std::error::Error + Send + Sync + 'static>),
20    #[error("invalid ops columns: {0}")]
21    InvalidOpsColumns(Box<dyn std::error::Error + Send + Sync + 'static>),

23    LeftoverData,
24    #[error("error inflating document chunk ops: {0}")]
25    InflateDocument(Box<dyn std::error::Error + Send + Sync + 'static>),

30pub(crate) enum LoadedChanges<'a> {
31    /// All the data was succesfully loaded into a list of changes
32    Complete(Vec<Change>),
33    /// We only managed to load _some_ changes.
34    Partial {
35        /// The succesfully loaded changes
36        loaded: Vec<Change>,
37        /// The data which we were unable to parse
38        #[allow(dead_code)]
39        remaining: parse::Input<'a>,
40        /// The error encountered whilst trying to parse `remaining`
41        error: Error,
42    },
43}
44
45/// Attempt to Load all the chunks in `data`.
46///
47/// # Partial Loads
48///
49/// Automerge documents are encoded as one or more concatenated chunks. Each chunk containing one
50/// or more changes. This means it is possible to partially load corrupted data if the first `n`
51/// chunks are valid. This function returns a `LoadedChanges` which you can examine to determine if
52/// this is the case.
53#[instrument(skip(data))]
54pub(crate) fn load_changes<'a>(mut data: parse::Input<'a>) -> LoadedChanges<'a> {
55    let mut changes = Vec::new();
56    while !data.is_empty() {
57        let remaining = match load_next_change(data, &mut changes) {
58            Ok(d) => d,
59            Err(e) => {
60                return LoadedChanges::Partial {
61                    loaded: changes,
62                    remaining: data,
63                    error: e,
64                };
65            }
66        };
67        data = remaining.reset();
68    }
69    LoadedChanges::Complete(changes)
70}

72fn load_next_change<'a>(
73    data: parse::Input<'a>,
74    changes: &mut Vec<Change>,
75) -> Result<parse::Input<'a>, Error> {
76    let (remaining, chunk) = storage::Chunk::parse(data).map_err(|e| Error::Parse(Box::new(e)))?;
77    if !chunk.checksum_valid() {
78        return Err(Error::BadChecksum);
79    }
80    match chunk {
81        storage::Chunk::Document(d) => {
82            tracing::trace!("loading document chunk");
83            let new_changes = reconstruct_opset(&d, VerificationMode::DontCheck)
84                .map_err(|e| Error::InflateDocument(Box::new(e)))?
85                .changes;
86            changes.extend(new_changes);
87        }
88        storage::Chunk::Change(change) => {
89            tracing::trace!("loading change chunk");
90            let change = Change::new_from_unverified(change.into_owned(), None)
91                .map_err(|e| Error::InvalidChangeColumns(Box::new(e)))?;
92            #[cfg(debug_assertions)]
93            {
94                let loaded_ops = change.iter_ops().collect::<Vec<_>>();
95                tracing::trace!(actor=?change.actor_id(), num_ops=change.len(), ops=?loaded_ops, "loaded change");
96            }
97            #[cfg(not(debug_assertions))]
98            tracing::trace!(actor=?change.actor_id(), num_ops=change.len(), "loaded change");
99            changes.push(change);
100        }
101        storage::Chunk::CompressedChange(change, compressed) => {
102            tracing::trace!("loading compressed change chunk");
103            let change =
104                Change::new_from_unverified(change.into_owned(), Some(compressed.into_owned()))
105                    .map_err(|e| Error::InvalidChangeColumns(Box::new(e)))?;
106            changes.push(change);
107        }
108    };
109    Ok(remaining)
110}
