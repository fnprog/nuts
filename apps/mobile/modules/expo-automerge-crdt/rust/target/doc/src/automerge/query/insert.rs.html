12#[derive(Debug, Clone, PartialEq)]
13pub(crate) struct InsertNth<'a> {
14    list_state: ListState,
15    clock: Option<Clock>,
16    last_visible_key: Option<Key>,
17    candidates: Vec<Loc<'a>>,
18    marks: QueriedMarks<'a>,
19}

21#[derive(Debug, Clone, PartialEq)]
22enum QueriedMarks<'a> {
23    FromQuery(RichTextQueryState<'a>),
24    FromLastSeen(Arc<MarkSet>),
25}
26
27impl<'a> std::default::Default for QueriedMarks<'a> {
28    fn default() -> Self {
29        QueriedMarks::FromQuery(Default::default())
30    }

33#[derive(Clone, PartialEq)]
34struct Loc<'a> {
35    key: Key,
36    pos: usize,
37    id: Option<Op<'a>>,
38}

40impl<'a> std::fmt::Debug for Loc<'a> {
41    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
42        write!(
43            f,
44            "Loc {{ key: {:?}, pos: {:?}, id: {:?} }}",
45            self.key,
46            self.pos,
47            self.id.map(|i| (i.id(), i.action()))
48        )
49    }
50}

52impl<'a> Loc<'a> {
53    fn new(pos: usize, key: Key) -> Self {
54        Loc { key, pos, id: None }
55    }
56
57    fn mark(pos: usize, key: Key, op: Op<'a>) -> Self {
58        Loc {
59            key,
60            pos,
61            id: Some(op),
62        }
63    }
64
65    fn matches(&self, op: Op<'a>) -> bool {
66        self.id.map(|o| o.id()) == Some(&op.id().prev())
67    }
68}

70impl<'a> InsertNth<'a> {
71    pub(crate) fn new(target: usize, encoding: ListEncoding, clock: Option<Clock>) -> Self {
72        let list_state = ListState::new(encoding, target);
73        if target == 0 {
74            InsertNth {
75                list_state,
76                last_visible_key: None,
77                candidates: vec![Loc::new(0, Key::Seq(HEAD))],
78                clock,
79                marks: Default::default(),
80            }
81        } else {
82            InsertNth {
83                list_state,
84                last_visible_key: None,
85                candidates: vec![],
86                clock,
87                marks: Default::default(),
88            }
89        }
90    }
91
92    pub(crate) fn marks(&self, osd: &OpSetData) -> Option<Arc<MarkSet>> {
93        match self.marks {
94            QueriedMarks::FromQuery(ref state) => MarkSet::from_query_state(state, osd),
95            QueriedMarks::FromLastSeen(ref rt) => Some(rt.clone()),
96        }
97    }
98
99    pub(crate) fn pos(&self) -> usize {
100        self.candidates
101            .last()
102            .map(|loc| loc.pos)
103            .unwrap_or(self.list_state.pos())
104    }
105
106    pub(crate) fn key(&self) -> Result<Key, AutomergeError> {
107        self.candidates
108            .last()
109            .map(|loc| loc.key)
110            .or(self.last_visible_key)
111            .ok_or(AutomergeError::InvalidIndex(self.list_state.target()))
112    }
113
114    fn identify_valid_insertion_spot(&mut self, op: Op<'a>, key: &Key) {
115        if !self.list_state.done() {
116            return;
117        }
118
119        // first insert we see after list_state.done()
120        if op.insert() && self.candidates.is_empty() && self.last_visible_key.is_some() {
121            if let Some(key) = self.last_visible_key {
122                self.candidates.push(Loc::new(self.list_state.pos(), key))
123            }
124        }
125
126        // sticky marks
127        if !self.candidates.is_empty() {
128            // if we find a begin/end pair - ignore them
129            if let OpType::MarkEnd(_) = op.action() {
130                if let Some(pos) = self.candidates.iter().position(|loc| loc.matches(op)) {
131                    // mark points between begin and end are invalid
132                    self.candidates.truncate(pos);
133                    return;
134                }
135            }
136            if matches!(
137                op.action(),
138                OpType::MarkBegin(true, _) | OpType::MarkEnd(false)
139            ) {
140                if let QueriedMarks::FromQuery(ref mut marks) = self.marks {
141                    marks.process(op, self.clock.as_ref());
142                }
143                self.candidates
144                    .push(Loc::mark(self.list_state.pos() + 1, *key, op));
145            }
146        }
147    }
148}

150impl<'a> TreeQuery<'a> for InsertNth<'a> {
151    fn can_shortcut_search(&mut self, tree: &'a OpTree, _osd: &'a OpSetData) -> bool {
152        if let Some(last) = &tree.last_insert {
153            if last.index + last.width == self.list_state.target() {
154                self.candidates.push(Loc::new(last.pos + 1, last.key));
155                if let Some(marks) = &last.marks {
156                    self.marks = QueriedMarks::FromLastSeen(marks.clone());
157                }
158                return true;
159            }
160        }
161        false
162    }
163
164    fn query_node(
165        &mut self,
166        child: &'a OpTreeNode,
167        index: &'a Index,
168        osd: &'a OpSetData,
169    ) -> QueryResult {
170        self.list_state.check_if_node_is_clean(index);
171        if self.clock.is_none() {
172            let marks = match &mut self.marks {
173                QueriedMarks::FromQuery(state) => Some(state),
174                QueriedMarks::FromLastSeen(_) => None,
175            };
176            self.list_state.process_node(child, index, osd, marks)
177        } else {
178            QueryResult::Descend
179        }
180    }
181
182    fn query_element(&mut self, op: Op<'a>) -> QueryResult {
183        if !self.list_state.done() {
184            if let QueriedMarks::FromQuery(ref mut marks) = self.marks {
185                marks.process(op, self.clock.as_ref());
186            }
187        }
188        let key = op.elemid_or_key();
189        let visible = op.visible_at(self.clock.as_ref());
190        self.identify_valid_insertion_spot(op, &key);
191        if visible {
192            if !self.candidates.is_empty() {
193                return QueryResult::Finish;
194            }
195            self.last_visible_key = Some(key);
196        }
197        self.list_state.process_op(op, key, visible);
198        QueryResult::Next
199    }
