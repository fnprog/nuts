55#[derive(Clone, Debug, PartialEq)]
56struct TextWidth {
57    width: usize,
58}
59
60impl TextWidth {
61    fn add_op(&mut self, op: Op<'_>) {
62        self.width += op.width(ListEncoding::Text);
63    }
64
65    fn remove_op(&mut self, op: Op<'_>) {
66        // Why are we using saturating_sub here? Shouldn't this always be greater than 0?
67        //
68        // In the case of objects which are _not_ `Text` we may end up subtracting more than the
69        // current width. This can happen if the elements in a list are `ScalarValue::str` and
70        // there are conflicting elements for the same index in the list. Like so:
71        //
72        // ```notrust
73        // [
74        //     "element",
75        //     ["conflict1", "conflict2_longer"],
76        //     "element"
77        // ]
78        // ```
79        //
80        // Where there are two conflicted elements at index 1
81        //
82        // in `Index::insert` and `Index::change_visibility` we add the width of the inserted op in
83        // utf8 and utf16 to the current width, but only if there was not a previous element for
84        // that index. Imagine that we encounter the "conflict1" op first, then we will add the
85        // length of 'conflict1' to the text widths. When 'conflict2_longer' is added we don't do
86        // anything because we've already seen an op for this index. Imagine that later we remove
87        // the `conflict2_longer` op, then we will end up subtracting the length of
88        // 'conflict2_longer' from the text widths, hence, `saturating_sub`. This isn't a problem
89        // because for non text objects we don't need the text widths to be accurate anyway.
90        //
91        // Really this is a sign that we should be tracking the type of the Index (List or Text) at
92        // the type level, but for now we just look the other way.
93        self.width = self.width.saturating_sub(op.width(ListEncoding::Text));
94    }
95
96    fn merge(&mut self, other: &TextWidth) {
97        self.width += other.width;
98    }
99}

102pub(crate) struct Index {
103    /// The map of visible keys to the number of visible operations for that key.
104    visible: HashMap<Key, usize, FxBuildHasher>,
105    visible_text: TextWidth,
106    /// Set of opids found in this node and below.
107    ops: HashSet<OpId, FxBuildHasher>,
108    never_seen_puts: bool,
109    mark_begin: HashMap<OpId, MarkData, FxBuildHasher>,
110    mark_end: Vec<OpId>,
111    /// The ID of the last block in this index, if any
112    pub(crate) block: Option<OpId>,
113}

144    pub(crate) fn change_vis<'a>(
145        &mut self,
146        change_vis: ChangeVisibility<'a>,
147    ) -> ChangeVisibility<'a> {
148        let ChangeVisibility {
149            old_vis,
150            new_vis,
151            op,
152        } = change_vis;
153        let key = op.elemid_or_key();
154        match (old_vis, new_vis) {
155            (true, false) => match self.visible.get(&key).copied() {
156                Some(1) => {
157                    self.visible.remove(&key);
158                    self.visible_text.remove_op(op);
159                }
160                Some(n) => {
161                    self.visible.insert(key, n - 1);
162                }
163                None => panic!("remove overun in index"),
164            },
165            (false, true) => {
166                if let Some(n) = self.visible.get(&key) {
167                    self.visible.insert(key, n + 1);
168                } else {
169                    self.visible.insert(key, 1);
170                    self.visible_text.add_op(op);
171                }
172            }
173            _ => {}
174        }
175        change_vis
176    }

178    pub(crate) fn insert(&mut self, op: Op<'_>) {
179        self.never_seen_puts &= op.insert();
180
181        // opids
182        self.ops.insert(*op.id());
183
184        // marks
185        match op.action() {
186            OpType::MarkBegin(_, data) => {
187                self.mark_begin.insert(*op.id(), data.clone());
188            }
189            OpType::MarkEnd(_) => {
190                if self.mark_begin.remove(&op.id().prev()).is_none() {
191                    self.mark_end.push(*op.id())
192                }
193            }
194            _ => {}
195        }
196
197        // visible ops
198        if op.visible() {
199            let key = op.elemid_or_key();
200            if let Some(n) = self.visible.get(&key) {
201                self.visible.insert(key, n + 1);
202            } else {
203                self.visible.insert(key, 1);
204                self.visible_text.add_op(op);
205            }
206        }
207    }
208
209    pub(crate) fn remove(&mut self, op: Op<'_>) {
210        // op ids
211        self.ops.remove(op.id());
212
213        // marks
214        match op.action() {
215            OpType::MarkBegin(_, _) => {
216                self.mark_begin.remove(op.id());
217            }
218            OpType::MarkEnd(_) => {
219                self.mark_end.retain(|id| id != op.id());
220            }
221            _ => {}
222        }
223
224        // visible ops
225        if op.visible() {
226            let key = op.elemid_or_key();
227            match self.visible.get(&key).copied() {
228                Some(1) => {
229                    self.visible.remove(&key);
230                    self.visible_text.remove_op(op);
231                }
232                Some(n) => {
233                    self.visible.insert(key, n - 1);
234                }
235                None => panic!("remove overun in index"),
236            }
237        }
238    }
239
240    pub(crate) fn merge(&mut self, other: &Index) {
241        for id in &other.ops {
242            self.ops.insert(*id);
243        }
244        for (elem, other_len) in other.visible.iter() {
245            self.visible
246                .entry(*elem)
247                .and_modify(|len| *len += *other_len)
248                .or_insert(*other_len);
249        }
250        self.mark_begin.extend(other.mark_begin.clone()); // can I remove this clone?
251        self.mark_end.extend(&other.mark_end);
252        self.visible_text.merge(&other.visible_text);
253        self.block = other.block;
254        self.never_seen_puts &= other.never_seen_puts;
255    }
256}
