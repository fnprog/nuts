22#[derive(Debug, thiserror::Error)]
23pub(crate) enum Error {
24    #[error("a change referenced an actor index we couldn't find")]
25    MissingActor,
26    #[error("changes out of order")]
27    ChangesOutOfOrder,
28    #[error("missing change")]
29    MissingChange,
30    #[error("unable to read change osd: {0}")]
31    ReadChange(Box<dyn std::error::Error + Send + Sync + 'static>),
32    #[error("incorrect max op")]
33    IncorrectMaxOp,
34    #[error("missing ops")]
35    MissingOps,
36}

47impl<'a> ChangeCollector<'a> {
48    pub(crate) fn new<E: std::error::Error + Send + Sync + 'static, I>(
49        changes: I,
50    ) -> Result<ChangeCollector<'a>, Error>
51    where
52        I: IntoIterator<Item = Result<ChangeMetadata<'a>, E>>,
53    {
54        let mut changes_by_actor: HashMap<usize, Vec<PartialChange<'_>>, FxBuildHasher> =
55            HashMap::default();
56        for (index, change) in changes.into_iter().enumerate() {
57            tracing::trace!(?change, "importing change osd");
58            let change = change.map_err(|e| Error::ReadChange(Box::new(e)))?;
59            let actor_changes = changes_by_actor.entry(change.actor).or_default();
60            if let Some(prev) = actor_changes.last() {
61                // Note that we allow max_op to be equal to the previous max_op in case the
62                // previous change had no ops (which is permitted)
63                if prev.max_op > change.max_op {
64                    return Err(Error::ChangesOutOfOrder);
65                }
66            }
67            actor_changes.push(PartialChange {
68                index,
69                deps: change.deps,
70                actor: change.actor,
71                seq: change.seq,
72                timestamp: change.timestamp,
73                max_op: change.max_op,
74                message: change.message,
75                extra_bytes: change.extra,
76                ops: Vec::new(),
77            })
78        }
79        let num_changes: usize = changes_by_actor.values().map(|v| v.len()).sum();
80        tracing::trace!(num_changes, "change collection context created");
81        Ok(ChangeCollector { changes_by_actor })
82    }
83
84    #[instrument(skip(self))]
85    pub(crate) fn collect(&mut self, opid: OpId, idx: OpIdx) -> Result<(), Error> {
86        let actor_changes = self
87            .changes_by_actor
88            .get_mut(&opid.actor())
89            .ok_or_else(|| {
90                tracing::error!(missing_actor = opid.actor(), "missing actor for op");
91                Error::MissingActor
92            })?;
93        let change_index = actor_changes.partition_point(|c| c.max_op < opid.counter());
94        let change = actor_changes.get_mut(change_index).ok_or_else(|| {
95            tracing::error!(missing_change_index = change_index, "missing change for op");
96            Error::MissingChange
97        })?;
98        change.ops.push(idx);
99        Ok(())
100    }
101
102    #[instrument(skip(self, osd))]
103    pub(crate) fn finish(self, osd: &OpSetData) -> Result<CollectedChanges<'static>, Error> {
104        let mut changes_in_order =
105            Vec::with_capacity(self.changes_by_actor.values().map(|c| c.len()).sum());
106        for (_, changes) in self.changes_by_actor {
107            let mut seq = None;
108            for change in changes {
109                if let Some(seq) = seq {
110                    if seq != change.seq - 1 {
111                        return Err(Error::ChangesOutOfOrder);
112                    }
113                } else if change.seq != 1 {
114                    return Err(Error::ChangesOutOfOrder);
115                }
116                seq = Some(change.seq);
117                changes_in_order.push(change);
118            }
119        }
120        changes_in_order.sort_by_key(|c| c.index);
121
122        let mut hashes_by_index = HashMap::default();
123        let mut history = Vec::new();
124        let mut heads = BTreeSet::new();
125        for (index, change) in changes_in_order.into_iter().enumerate() {
126            let finished = change.finish(&hashes_by_index, osd)?;
127            let hash = finished.hash();
128            hashes_by_index.insert(index, hash);
129            for dep in finished.dependencies() {
130                heads.remove(dep);
131            }
132            heads.insert(hash);
133            history.push(finished.into_owned());
134        }
135
136        Ok(CollectedChanges { history, heads })
137    }
138}

140#[derive(Debug)]
141struct PartialChange<'a> {
142    index: usize,
143    deps: Vec<u64>,
144    actor: usize,
145    seq: u64,
146    max_op: u64,
147    timestamp: i64,
148    message: Option<smol_str::SmolStr>,
149    extra_bytes: Cow<'a, [u8]>,
150    ops: Vec<OpIdx>,
151}
152
153impl<'a> PartialChange<'a> {
154    /// # Panics
155    ///
156    /// * If any op references a property index which is not in `props`
157    /// * If any op references an actor index which is not in `actors`
158    #[instrument(skip(self, known_changes, osd))]
159    fn finish(
160        mut self,
161        known_changes: &HashMap<usize, ChangeHash, FxBuildHasher>,
162        osd: &OpSetData,
163    ) -> Result<StoredChange<'a, Verified>, Error> {
164        let deps_len = self.deps.len();
165        let mut deps = self.deps.into_iter().try_fold::<_, _, Result<_, Error>>(
166            Vec::with_capacity(deps_len),
167            |mut acc, dep| {
168                acc.push(known_changes.get(&(dep as usize)).cloned().ok_or_else(|| {
169                    tracing::error!(
170                        dependent_index = self.index,
171                        dep_index = dep,
172                        "could not find dependency"
173                    );
174                    Error::MissingChange
175                })?);
176                Ok(acc)
177            },
178        )?;
179        deps.sort();
180        let num_ops = self.ops.len() as u64;
181        self.ops.sort_by_key(|o| o.as_op(osd).id().counter());
182        let converted_ops = self.ops.iter().map(|idx| op_as_actor_id(idx.as_op(osd)));
183        let actor = osd
184            .actors
185            .safe_get(self.actor)
186            .ok_or_else(|| {
187                tracing::error!(actor_index = self.actor, "actor out of bounds");
188                Error::MissingActor
189            })?
190            .clone();
191
192        if num_ops > self.max_op {
193            return Err(Error::IncorrectMaxOp);
194        }
195
196        let change = match StoredChange::builder()
197            .with_dependencies(deps)
198            .with_actor(actor)
199            .with_seq(self.seq)
200            .with_start_op(NonZeroU64::new(self.max_op - num_ops + 1).ok_or(Error::MissingOps)?)
201            .with_timestamp(self.timestamp)
202            .with_message(self.message.map(|s| s.to_string()))
203            .with_extra_bytes(self.extra_bytes.into_owned())
204            .build(converted_ops)
205        {
206            Ok(s) => s,
207            Err(PredOutOfOrder) => {
208                // SAFETY: types::OpBuilder::preds is `types::OpIds` which ensures ops are always sorted
209                panic!("preds out of order");
210            }
211        };
212        #[cfg(not(debug_assertions))]
213        tracing::trace!(?change, hash=?change.hash(), "collected change");
214        #[cfg(debug_assertions)]
215        {
216            tracing::trace!(?change, ops=?self.ops, hash=?change.hash(), "collected change");
217        }
218        Ok(change)
219    }
