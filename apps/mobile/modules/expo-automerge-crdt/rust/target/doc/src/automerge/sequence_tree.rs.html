15#[derive(Clone, Debug, PartialEq)]
16struct SequenceTreeNode<T> {
17    elements: Vec<T>,
18    children: Vec<SequenceTreeNode<T>>,
19    length: usize,

22impl<T> SequenceTreeInternal<T>
23where
24    T: Clone + Debug,
25{
26    /// Construct a new, empty, sequence.
27    pub fn new() -> Self {
28        Self { root_node: None }
29    }
30
31    /// Get the length of the sequence.
32    pub fn len(&self) -> usize {
33        self.root_node.as_ref().map_or(0, |n| n.len())
34    }
35
36    /// Create an iterator through the sequence.
37    pub fn iter(&self) -> Iter<'_, T> {
38        Iter {
39            inner: self,
40            index: 0,
41        }
42    }
43
44    /// Insert the `element` into the sequence at `index`.
45    ///
46    /// # Panics
47    ///
48    /// Panics if `index > len`.
49    pub fn insert(&mut self, index: usize, element: T) {
50        let old_len = self.len();
51        if let Some(root) = self.root_node.as_mut() {
52            #[cfg(debug_assertions)]
53            root.check();
54
55            if root.is_full() {
56                let original_len = root.len();
57                let new_root = SequenceTreeNode::new();
58
59                // move new_root to root position
60                let old_root = mem::replace(root, new_root);
61
62                root.length += old_root.len();
63                root.children.push(old_root);
64                root.split_child(0);
65
66                assert_eq!(original_len, root.len());
67
68                // after splitting the root has one element and two children, find which child the
69                // index is in
70                let first_child_len = root.children[0].len();
71                let (child, insertion_index) = if first_child_len < index {
72                    (&mut root.children[1], index - (first_child_len + 1))
73                } else {
74                    (&mut root.children[0], index)
75                };
76                root.length += 1;
77                child.insert_into_non_full_node(insertion_index, element)
78            } else {
79                root.insert_into_non_full_node(index, element)
80            }
81        } else {
82            self.root_node = Some(SequenceTreeNode {
83                elements: vec![element],
84                children: Vec::new(),
85                length: 1,
86            })
87        }
88        assert_eq!(self.len(), old_len + 1, "{:#?}", self);
89    }
90
91    /// Push the `element` onto the back of the sequence.
92    pub fn push(&mut self, element: T) {
93        let l = self.len();
94        self.insert(l, element)
95    }
96
97    /// Get the `element` at `index` in the sequence.
98    pub fn get(&self, index: usize) -> Option<&T> {
99        self.root_node.as_ref().and_then(|n| n.get(index))
100    }
101
102    /// Get the mut ref for the `element` at `index` in the sequence.
103    pub fn get_mut(&mut self, index: usize) -> Option<&mut T> {
104        self.root_node.as_mut().and_then(|n| n.get_mut(index))
105    }
106
107    /// Removes the element at `index` from the sequence.
108    ///
109    /// # Panics
110    ///
111    /// Panics if `index` is out of bounds.
112    pub fn remove(&mut self, index: usize) -> T {
113        if let Some(root) = self.root_node.as_mut() {
114            #[cfg(debug_assertions)]
115            let len = root.check();
116            let old = root.remove(index);
117
118            if root.elements.is_empty() {
119                if root.is_leaf() {
120                    self.root_node = None;
121                } else {
122                    self.root_node = Some(root.children.remove(0));
123                }
124            }
125
126            #[cfg(debug_assertions)]
127            debug_assert_eq!(len, self.root_node.as_ref().map_or(0, |r| r.check()) + 1);
128            old
129        } else {
130            panic!("remove from empty tree")
131        }
132    }
133}

135impl<T> SequenceTreeNode<T>
136where
137    T: Clone + Debug,
138{
139    fn new() -> Self {
140        Self {
141            elements: Vec::new(),
142            children: Vec::new(),
143            length: 0,
144        }
145    }
146
147    fn len(&self) -> usize {
148        self.length
149    }
150
151    fn is_leaf(&self) -> bool {
152        self.children.is_empty()
153    }
154
155    fn is_full(&self) -> bool {
156        self.elements.len() >= 2 * B - 1
157    }
158
159    /// Returns the child index and the given index adjusted for the cumulative index before that
160    /// child.
161    fn find_child_index(&self, index: usize) -> (usize, usize) {
162        let mut cumulative_len = 0;
163        for (child_index, child) in self.children.iter().enumerate() {
164            if cumulative_len + child.len() >= index {
165                return (child_index, index - cumulative_len);
166            } else {
167                cumulative_len += child.len() + 1;
168            }
169        }
170        panic!("index not found in node")
171    }
172
173    fn insert_into_non_full_node(&mut self, index: usize, element: T) {
174        assert!(!self.is_full());
175        if self.is_leaf() {
176            self.length += 1;
177            self.elements.insert(index, element);
178        } else {
179            let (child_index, sub_index) = self.find_child_index(index);
180            let child = &mut self.children[child_index];
181
182            if child.is_full() {
183                self.split_child(child_index);
184
185                // child structure has changed so we need to find the index again
186                let (child_index, sub_index) = self.find_child_index(index);
187                let child = &mut self.children[child_index];
188                child.insert_into_non_full_node(sub_index, element);
189            } else {
190                child.insert_into_non_full_node(sub_index, element);
191            }
192            self.length += 1;
193        }
194    }
195
196    // A utility function to split the child `full_child_index` of this node
197    // Note that `full_child_index` must be full when this function is called.
198    fn split_child(&mut self, full_child_index: usize) {
199        let original_len_self = self.len();
200
201        // Create a new node which is going to store (B-1) keys
202        // of the full child.
203        let mut successor_sibling = SequenceTreeNode::new();
204
205        let full_child = &mut self.children[full_child_index];
206        let original_len = full_child.len();
207        assert!(full_child.is_full());
208
209        successor_sibling.elements = full_child.elements.split_off(B);
210
211        if !full_child.is_leaf() {
212            successor_sibling.children = full_child.children.split_off(B);
213        }
214
215        let middle = full_child.elements.pop().unwrap();
216
217        full_child.length =
218            full_child.elements.len() + full_child.children.iter().map(|c| c.len()).sum::<usize>();
219
220        successor_sibling.length = successor_sibling.elements.len()
221            + successor_sibling
222                .children
223                .iter()
224                .map(|c| c.len())
225                .sum::<usize>();
226
227        let z_len = successor_sibling.len();
228
229        let full_child_len = full_child.len();
230
231        self.children
232            .insert(full_child_index + 1, successor_sibling);
233
234        self.elements.insert(full_child_index, middle);
235
236        assert_eq!(full_child_len + z_len + 1, original_len, "{:#?}", self);
237
238        assert_eq!(original_len_self, self.len());
239    }
240
241    fn remove_from_leaf(&mut self, index: usize) -> T {
242        self.length -= 1;
243        self.elements.remove(index)
244    }
245
246    fn remove_element_from_non_leaf(&mut self, index: usize, element_index: usize) -> T {
247        self.length -= 1;
248        if self.children[element_index].elements.len() >= B {
249            let total_index = self.cumulative_index(element_index);
250            // recursively delete index - 1 in predecessor_node
251            let predecessor = self.children[element_index].remove(index - 1 - total_index);
252            // replace element with that one
253            mem::replace(&mut self.elements[element_index], predecessor)
254        } else if self.children[element_index + 1].elements.len() >= B {
255            // recursively delete index + 1 in successor_node
256            let total_index = self.cumulative_index(element_index + 1);
257            let successor = self.children[element_index + 1].remove(index + 1 - total_index);
258            // replace element with that one
259            mem::replace(&mut self.elements[element_index], successor)
260        } else {
261            let middle_element = self.elements.remove(element_index);
262            let successor_child = self.children.remove(element_index + 1);
263            self.children[element_index].merge(middle_element, successor_child);
264
265            let total_index = self.cumulative_index(element_index);
266            self.children[element_index].remove(index - total_index)
267        }
268    }
269
270    fn cumulative_index(&self, child_index: usize) -> usize {
271        self.children[0..child_index]
272            .iter()
273            .map(|c| c.len() + 1)
274            .sum()
275    }
276
277    fn remove_from_internal_child(&mut self, index: usize, mut child_index: usize) -> T {
278        if self.children[child_index].elements.len() < B
279            && if child_index > 0 {
280                self.children[child_index - 1].elements.len() < B
281            } else {
282                true
283            }
284            && if child_index + 1 < self.children.len() {
285                self.children[child_index + 1].elements.len() < B
286            } else {
287                true
288            }
289        {
290            // if the child and its immediate siblings have B-1 elements merge the child
291            // with one sibling, moving an element from this node into the new merged node
292            // to be the median
293
294            if child_index > 0 {
295                let middle = self.elements.remove(child_index - 1);
296
297                // use the predessor sibling
298                let successor = self.children.remove(child_index);
299                child_index -= 1;
300
301                self.children[child_index].merge(middle, successor);
302            } else {
303                let middle = self.elements.remove(child_index);
304
305                // use the sucessor sibling
306                let successor = self.children.remove(child_index + 1);
307
308                self.children[child_index].merge(middle, successor);
309            }
310        } else if self.children[child_index].elements.len() < B {
311            if child_index > 0
312                && self
313                    .children
314                    .get(child_index - 1)
315                    .map_or(false, |c| c.elements.len() >= B)
316            {
317                let last_element = self.children[child_index - 1].elements.pop().unwrap();
318                assert!(!self.children[child_index - 1].elements.is_empty());
319                self.children[child_index - 1].length -= 1;
320
321                let parent_element =
322                    mem::replace(&mut self.elements[child_index - 1], last_element);
323
324                self.children[child_index]
325                    .elements
326                    .insert(0, parent_element);
327                self.children[child_index].length += 1;
328
329                if let Some(last_child) = self.children[child_index - 1].children.pop() {
330                    self.children[child_index - 1].length -= last_child.len();
331                    self.children[child_index].length += last_child.len();
332                    self.children[child_index].children.insert(0, last_child);
333                }
334            } else if self
335                .children
336                .get(child_index + 1)
337                .map_or(false, |c| c.elements.len() >= B)
338            {
339                let first_element = self.children[child_index + 1].elements.remove(0);
340                self.children[child_index + 1].length -= 1;
341
342                assert!(!self.children[child_index + 1].elements.is_empty());
343
344                let parent_element = mem::replace(&mut self.elements[child_index], first_element);
345
346                self.children[child_index].length += 1;
347                self.children[child_index].elements.push(parent_element);
348
349                if !self.children[child_index + 1].is_leaf() {
350                    let first_child = self.children[child_index + 1].children.remove(0);
351                    self.children[child_index + 1].length -= first_child.len();
352                    self.children[child_index].length += first_child.len();
353
354                    self.children[child_index].children.push(first_child);
355                }
356            }
357        }
358        self.length -= 1;
359        let total_index = self.cumulative_index(child_index);
360        self.children[child_index].remove(index - total_index)
361    }
362
363    fn check(&self) -> usize {
364        let l = self.elements.len() + self.children.iter().map(|c| c.check()).sum::<usize>();
365        assert_eq!(self.len(), l, "{:#?}", self);
366
367        l
368    }
369
370    fn remove(&mut self, index: usize) -> T {
371        let original_len = self.len();
372        if self.is_leaf() {
373            let v = self.remove_from_leaf(index);
374            assert_eq!(original_len, self.len() + 1);
375            debug_assert_eq!(self.check(), self.len());
376            v
377        } else {
378            let mut total_index = 0;
379            for (child_index, child) in self.children.iter().enumerate() {
380                match (total_index + child.len()).cmp(&index) {
381                    Ordering::Less => {
382                        // should be later on in the loop
383                        total_index += child.len() + 1;
384                        continue;
385                    }
386                    Ordering::Equal => {
387                        let v = self.remove_element_from_non_leaf(
388                            index,
389                            min(child_index, self.elements.len() - 1),
390                        );
391                        assert_eq!(original_len, self.len() + 1);
392                        debug_assert_eq!(self.check(), self.len());
393                        return v;
394                    }
395                    Ordering::Greater => {
396                        let v = self.remove_from_internal_child(index, child_index);
397                        assert_eq!(original_len, self.len() + 1);
398                        debug_assert_eq!(self.check(), self.len());
399                        return v;
400                    }
401                }
402            }
403            panic!(
404                "index not found to remove {} {} {} {}",
405                index,
406                total_index,
407                self.len(),
408                self.check()
409            );
410        }
411    }
412
413    fn merge(&mut self, middle: T, successor_sibling: SequenceTreeNode<T>) {
414        self.elements.push(middle);
415        self.elements.extend(successor_sibling.elements);
416        self.children.extend(successor_sibling.children);
417        self.length += successor_sibling.length + 1;
418        assert!(self.is_full());
419    }
420
421    fn get(&self, index: usize) -> Option<&T> {
422        if self.is_leaf() {
423            return self.elements.get(index);
424        } else {
425            let mut cumulative_len = 0;
426            for (child_index, child) in self.children.iter().enumerate() {
427                match (cumulative_len + child.len()).cmp(&index) {
428                    Ordering::Less => {
429                        cumulative_len += child.len() + 1;
430                    }
431                    Ordering::Equal => return self.elements.get(child_index),
432                    Ordering::Greater => {
433                        return child.get(index - cumulative_len);
434                    }
435                }
436            }
437        }
438        None
439    }
440
441    fn get_mut(&mut self, index: usize) -> Option<&mut T> {
442        if self.is_leaf() {
443            return self.elements.get_mut(index);
444        } else {
445            let mut cumulative_len = 0;
446            for (child_index, child) in self.children.iter_mut().enumerate() {
447                match (cumulative_len + child.len()).cmp(&index) {
448                    Ordering::Less => {
449                        cumulative_len += child.len() + 1;
450                    }
451                    Ordering::Equal => return self.elements.get_mut(child_index),
452                    Ordering::Greater => {
453                        return child.get_mut(index - cumulative_len);
454                    }
455                }
456            }
457        }
458        None
459    }
