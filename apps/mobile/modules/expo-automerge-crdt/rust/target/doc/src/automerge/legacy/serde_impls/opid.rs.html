1use std::str::FromStr;

7impl<'de> Deserialize<'de> for OpId {
8    fn deserialize<D>(deserializer: D) -> Result<Self, D::Error>
9    where
10        D: Deserializer<'de>,
11    {
12        let s = String::deserialize(deserializer)?;
13        OpId::from_str(&s)
14            .map_err(|_| de::Error::invalid_value(de::Unexpected::Str(&s), &"A valid OpID"))
15    }
16}
17
18impl Serialize for OpId {
19    fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
20    where
21        S: Serializer,
22    {
23        serializer.serialize_str(self.to_string().as_str())
24    }
