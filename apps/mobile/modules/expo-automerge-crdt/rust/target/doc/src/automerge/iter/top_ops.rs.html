64impl<'a> Iterator for TopOpsInner<'a> {
65    type Item = TopOp<'a>;
66
67    fn next(&mut self) -> Option<Self::Item> {
68        let mut result_op = None;
69        loop {
70            if let Some(op) = self.iter.next() {
71                let key = op.elemid_or_key();
72                let visible = op.visible_at(self.clock.as_ref());
73                match &self.clock {
74                    Some(c) if c.covers(op.id()) => {
75                        self.marks.process(*op.id(), op.action(), self.iter.osd);
76                    }
77                    _ => {}
78                }
79                match &self.key {
80                    Some(k) if k == &key => {
81                        if visible {
82                            self.last_op = Some((self.pos, op, self.marks.current().cloned()));
83                            self.num_ops += 1;
84                        }
85                    }
86                    Some(_) => {
87                        result_op = self.last_op.take().map(|(_op_pos, op, marks)| (op, marks));
88                        if visible {
89                            self.last_op = Some((self.pos, op, self.marks.current().cloned()));
90                            self.num_ops = 1;
91                        } else {
92                            self.num_ops = 0;
93                        }
94                        self.key = Some(key);
95                        self.start_pos = self.pos;
96                    }
97                    None => {
98                        self.key = Some(key);
99                        self.start_pos = self.pos;
100                        if visible {
101                            self.last_op = Some((self.pos, op, self.marks.current().cloned()));
102                            self.num_ops = 1;
103                        } else {
104                            self.num_ops = 0;
105                        }
106                    }
107                }
108                self.pos += 1;
109                if result_op.is_some() {
110                    break;
111                }
112            } else {
113                result_op = self.last_op.take().map(|(_op_pos, op, marks)| (op, marks));
114                break;
115            }
116        }
117        result_op.map(|(op, marks)| TopOp {
118            op,
119            conflict: self.num_ops > 1,
120            marks,
121        })
122    }
