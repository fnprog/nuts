3use crate::exid::ExId;

7/// A way of mutating a document within a single change.
8pub trait Transactable: ReadDoc {
9    /// Get the number of pending operations in this transaction.
10    fn pending_ops(&self) -> usize;
11
12    /// Set the value of property `P` to value `V` in object `obj`.
13    ///
14    /// # Errors
15    ///
16    /// This will return an error if
17    /// - The object does not exist
18    /// - The key is the wrong type for the object
19    /// - The key does not exist in the object
20    fn put<O: AsRef<ExId>, P: Into<Prop>, V: Into<ScalarValue>>(
21        &mut self,
22        obj: O,
23        prop: P,
24        value: V,
25    ) -> Result<(), AutomergeError>;
26
27    /// Set the value of property `P` to the new object `V` in object `obj`.
28    ///
29    /// # Returns
30    ///
31    /// The id of the object which was created.
32    ///
33    /// # Errors
34    ///
35    /// This will return an error if
36    /// - The object does not exist
37    /// - The key is the wrong type for the object
38    /// - The key does not exist in the object
39    fn put_object<O: AsRef<ExId>, P: Into<Prop>>(
40        &mut self,
41        obj: O,
42        prop: P,
43        object: ObjType,
44    ) -> Result<ExId, AutomergeError>;
45
46    /// Insert a value into a list at the given index.
47    fn insert<O: AsRef<ExId>, V: Into<ScalarValue>>(
48        &mut self,
49        obj: O,
50        index: usize,
51        value: V,
52    ) -> Result<(), AutomergeError>;
53
54    /// Insert an object into a list at the given index.
55    fn insert_object<O: AsRef<ExId>>(
56        &mut self,
57        obj: O,
58        index: usize,
59        object: ObjType,
60    ) -> Result<ExId, AutomergeError>;
61
62    /// Increment the counter at the prop in the object by `value`.
63    fn increment<O: AsRef<ExId>, P: Into<Prop>>(
64        &mut self,
65        obj: O,
66        prop: P,
67        value: i64,
68    ) -> Result<(), AutomergeError>;
69
70    /// Delete the value at prop in the object.
71    fn delete<O: AsRef<ExId>, P: Into<Prop>>(
72        &mut self,
73        obj: O,
74        prop: P,
75    ) -> Result<(), AutomergeError>;
76
77    /// replace a section of a list. If `del` is positive then N values
78    /// are deleted after position `pos` and the new values inserted. If
79    /// it is negative then N values are deleted before position `pos` instead.
80    fn splice<O: AsRef<ExId>, V: IntoIterator<Item = ScalarValue>>(
81        &mut self,
82        obj: O,
83        pos: usize,
84        del: isize,
85        vals: V,
86    ) -> Result<(), AutomergeError>;
87
88    /// Like [`Self::splice`] but for text.
89    fn splice_text<O: AsRef<ExId>>(
90        &mut self,
91        obj: O,
92        pos: usize,
93        del: isize,
94        text: &str,
95    ) -> Result<(), AutomergeError>;
96
97    /// Mark a sequence
98    fn mark<O: AsRef<ExId>>(
99        &mut self,
100        obj: O,
101        mark: Mark<'_>,
102        expand: ExpandMark,
103    ) -> Result<(), AutomergeError>;
104
105    /// Remove a Mark from a sequence
106    fn unmark<O: AsRef<ExId>>(
107        &mut self,
108        obj: O,
109        key: &str,
110        start: usize,
111        end: usize,
112        expand: ExpandMark,
113    ) -> Result<(), AutomergeError>;
114
115    /// Insert a block marker into the text object `obj` at the given index.
116    ///
117    /// # Returns
118    ///
119    /// The ID of the new block marker. The block marker is a plain old map so you can use all the
120    /// normal methods of modifying a map to interact with it.
121    fn split_block<O>(&mut self, obj: O, index: usize) -> Result<ExId, AutomergeError>
122    where
123        O: AsRef<ExId>;
124
125    /// Delete a block marker at `index` from the text object `obj`.
126    fn join_block<O: AsRef<ExId>>(&mut self, text: O, index: usize) -> Result<(), AutomergeError>;
127
128    /// Replace a block marker at `index` in `obj` with a new marker and return the ID of the new
129    /// marker
130    fn replace_block<O>(&mut self, text: O, index: usize) -> Result<ExId, AutomergeError>
131    where
132        O: AsRef<ExId>;
133
134    /// Update the blocks and text in a text object
135    ///
136    /// This performs a diff against the current state of both the text and the block markers in a
137    /// text object and attempts to perform a reasonably minimal set of operations to update the
138    /// document to match the new text.
139    fn update_spans<'a, O: AsRef<ExId>, I: IntoIterator<Item = BlockOrText<'a>>>(
140        &mut self,
141        text: O,
142        new_text: I,
143    ) -> Result<(), AutomergeError>;
144
145    /// The heads this transaction will be based on
146    fn base_heads(&self) -> Vec<ChangeHash>;
147
148    /// Update the value of a string
149    ///
150    /// This will calculate a diff between the current value and the new value and
151    /// then convert that diff into calls to {@link splice}. This will produce results
152    /// which don't merge as well as directly capturing the user input actions, but
153    /// sometimes it's not possible to capture user input and this is the best you
154    /// can do.
155    fn update_text<S: AsRef<str>>(&mut self, obj: &ExId, new_text: S)
156        -> Result<(), AutomergeError>;
157
158    fn update_object<O: AsRef<ExId>>(
159        &mut self,
160        obj: O,
161        new_value: &crate::hydrate::Value,
162    ) -> Result<(), crate::error::UpdateObjectError>;
163}

165#[derive(Debug, PartialEq, Clone)]
166pub enum BlockOrText<'a> {
167    Block(crate::hydrate::Map),
168    Text(Cow<'a, str>),
169}
