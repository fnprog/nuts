103    pub(crate) fn log_patches(
104        &self,
105        obj: &ObjMeta,
106        op: Op<'_>,
107        pred: &OpIds,
108        doc: &Automerge,
109        patch_log: &mut PatchLog,
110    ) {
111        if op.insert() {
112            if op.is_mark() {
113                if let OpType::MarkEnd(_) = op.action() {
114                    let q = doc.ops().search(
115                        &obj.id,
116                        query::SeekMark::new(
117                            op.id().prev(),
118                            self.pos,
119                            patch_log.text_rep().encoding(obj.typ),
120                        ),
121                    );
122                    for mark in q.finish() {
123                        let index = mark.start;
124                        let len = mark.len();
125                        let marks = mark.into_mark_set();
126                        patch_log.mark(obj.id, index, len, &marks);
127                    }
128                }
129            // TODO - move this into patch_log()
130            } else if obj.typ == ObjType::Text && !op.action().is_block() {
131                patch_log.splice(obj.id, self.index, op.as_str(), self.marks.clone());
132            } else {
133                patch_log.insert(obj.id, self.index, op.value().into(), *op.id(), false);
134            }
135            return;
136        }
137
138        let key: Prop = match *op.key() {
139            Key::Map(i) => doc.ops().osd.props[i].clone().into(),
140            Key::Seq(_) => self.index.into(),
141        };
142
143        if op.is_delete() {
144            match (self.before, self.overwritten, self.after) {
145                (None, Some(over), None) => match key {
146                    Prop::Map(k) => patch_log.delete_map(obj.id, &k),
147                    Prop::Seq(index) => patch_log.delete_seq(
148                        obj.id,
149                        index,
150                        over.width(patch_log.text_rep().encoding(obj.typ)),
151                    ),
152                },
153                (Some(before), Some(_), None) => {
154                    let conflict = self.num_before > 1;
155                    patch_log.put(
156                        obj.id,
157                        &key,
158                        before.value().into(),
159                        *before.id(),
160                        conflict,
161                        true,
162                    );
163                }
164                _ => { /* do nothing */ }
165            }
166        } else if let Some(value) = op.get_increment_value() {
167            if self.after.is_none() {
168                if let Some(counter) = self.overwritten {
169                    if pred.overwrites(counter.id()) {
170                        patch_log.increment(obj.id, &key, value, *op.id());
171                    }
172                }
173            }
174        } else {
175            let conflict = self.before.is_some();
176            if op.is_list_op()
177                && self.overwritten.is_none()
178                && self.before.is_none()
179                && self.after.is_none()
180            {
181                patch_log.insert(obj.id, self.index, op.value().into(), *op.id(), conflict);
182            } else if self.after.is_some() {
183                if self.before.is_none() {
184                    patch_log.flag_conflict(obj.id, &key);
185                }
186            } else {
187                patch_log.put(obj.id, &key, op.value().into(), *op.id(), conflict, false);
188            }
189        }
190    }
191}

214    pub(crate) fn found_op_without_patch_log(
215        &self,
216        osd: &OpSetData,
217        op: Op<'_>,
218        pred: &OpIds,
219        mut pos: usize,
220    ) -> FoundOpWithoutPatchLog {
221        let mut iter = self.iter();
222        let mut succ = vec![];
223        let mut next = iter.nth(pos);
224        while let Some(idx) = next {
225            let e = idx.as_op(osd);
226            if e.elemid_or_key() != op.elemid_or_key() {
227                break;
228            }
229
230            if e.lamport_cmp(*op.id()) == Ordering::Greater {
231                break;
232            }
233
234            if pred.overwrites(e.id()) {
235                succ.push(pos);
236            }
237
238            pos += 1;
239            next = iter.next();
240        }
241
242        FoundOpWithoutPatchLog { pos, succ }
243    }

245    pub(crate) fn found_op_with_patch_log<'a>(
246        &'a self,
247        osd: &'a OpSetData,
248        op: Op<'a>,
249        pred: &OpIds,
250        mut pos: usize,
251        index: usize,
252        marks: Option<Arc<MarkSet>>,
253    ) -> FoundOpWithPatchLog<'a> {
254        let mut iter = self.iter();
255        let mut found = None;
256        let mut before = None;
257        let mut num_before = 0;
258        let mut overwritten = None;
259        let mut after = None;
260        let mut succ = vec![];
261        let mut next = iter.nth(pos);
262        while let Some(idx) = next {
263            let e = idx.as_op(osd);
264            if e.elemid_or_key() != op.elemid_or_key() {
265                break;
266            }
267
268            if found.is_none() && e.lamport_cmp(*op.id()) == Ordering::Greater {
269                found = Some(pos);
270            }
271
272            if pred.overwrites(e.id()) {
273                succ.push(pos);
274
275                if e.visible() {
276                    overwritten = Some(e);
277                }
278            } else if e.visible() {
279                if found.is_none() && overwritten.is_none() {
280                    before = Some(e);
281                    num_before += 1;
282                } else {
283                    after = Some(e);
284                }
285            }
286
287            pos += 1;
288            next = iter.next();
289        }
290
291        pos = found.unwrap_or(pos);
292
293        FoundOpWithPatchLog {
294            before,
295            num_before,
296            after,
297            overwritten,
298            succ,
299            pos,
300            index,
301            marks,
302        }
303    }

305    pub(crate) fn seek_map_op<'a>(
306        &'a self,
307        op: Op<'a>,
308        clock: Option<&Clock>,
309        osd: &'a OpSetData,
310    ) -> Option<FoundOpId<'a>> {
311        let pos = self.binary_search_by(osd, |o| o.key_cmp(op.key()).then_with(|| o.cmp(&op)));
312        let mut iter = self.iter();
313        let op2 = iter.nth(pos).map(|idx| idx.as_op(osd))?;
314        assert_eq!(op, op2);
315        let index = 0;
316        for e in iter.map(|idx| idx.as_op(osd)) {
317            if e.elemid_or_key() != op.elemid_or_key() {
318                break;
319            }
320
321            if e.visible_at(clock) {
322                return Some(FoundOpId {
323                    op,
324                    index,
325                    visible: false,
326                });
327            }
328        }
329        Some(FoundOpId {
330            op,
331            index,
332            visible: op.visible_at(clock),
333        })
334    }

387        op: Op<'a>,
388        pred: &OpIds,
389        encoding: ListEncoding,
390        osd: &'a OpSetData,

400            self.found_op_with_patch_log(osd, op, pred, pos, 0, None)

433    pub(crate) fn seek_ops_by_map_key<'a>(
434        &'a self,
435        osd: &'a OpSetData,
436        key_name: String,
437        clock: Option<&Clock>,
438    ) -> Option<OpsFound<'a>> {
439        let key = Key::Map(osd.props.lookup(&key_name)?);
440        let mut pos = self.binary_search_by(osd, |o| o.key_cmp(&key));
441        let mut iter = self.iter();
442        let mut next = iter.nth(pos);
443        let mut ops = vec![];
444        let mut ops_pos = vec![];
445        while let Some(op) = next {
446            let op = op.as_op(osd);
447            match op.key_cmp(&key) {
448                Ordering::Greater => {
449                    break;
450                }
451                Ordering::Equal if op.visible_at(clock) => {
452                    ops.push(op);
453                    ops_pos.push(pos);
454                }
455                _ => {}
456            }
457            pos += 1;
458            next = iter.next();
459        }
460        Some(OpsFound {
461            ops,
462            ops_pos,
463            end_pos: pos,
464        })
465    }

483    fn binary_search_by<F>(&self, osd: &OpSetData, f: F) -> usize
484    where
485        F: Fn(Op<'_>) -> Ordering,
486    {
487        let mut right = self.len();
488        let mut left = 0;
489        while left < right {
490            let seq = (left + right) / 2;
491            if f(self.get(seq).unwrap().as_op(osd)) == Ordering::Less {
492                left = seq + 1;
493            } else {
494                right = seq;
495            }
496        }
497        left
498    }

500    pub(crate) fn search<'a, 'b: 'a, Q>(&'b self, mut query: Q, osd: &'a OpSetData) -> Q
501    where
502        Q: TreeQuery<'a>,
503    {
504        self.root_node.as_ref().map(|root| {
505            if let Some(index) = root.index.as_ref() {
506                match query.query_node(root, index, osd) {
507                    QueryResult::Descend => root.search(&mut query, osd),
508                    _ => true,
509                }
510            } else {
511                // the only thing using this branch is rollback and it has a rewrite coming soon
512                root.search(&mut query, osd)
513            }
514        });
515        query
516    }

523    /// Insert the `element` into the sequence at `index`.
524    ///
525    /// # Panics
526    ///
527    /// Panics if `index > len`.
528    pub(crate) fn insert(&mut self, index: usize, element: OpIdx, osd: &OpSetData) {
529        assert!(
530            index <= self.len(),
531            "tried to insert at {} but len is {}",
532            index,
533            self.len()
534        );
535
536        let old_len = self.len();
537        if let Some(root) = self.root_node.as_mut() {
538            #[cfg(debug_assertions)]
539            root.check();
540
541            if root.is_full() {
542                let original_len = root.len();
543                let new_root = OpTreeNode::new(root.index.is_some());
544
545                // move new_root to root position
546                let old_root = mem::replace(root, new_root);
547
548                root.length += old_root.len();
549                root.index.clone_from(&old_root.index);
550                root.children.push(old_root);
551                root.split_child(0, osd);
552
553                assert_eq!(original_len, root.len());
554
555                // after splitting the root has one element and two children, find which child the
556                // index is in
557                let first_child_len = root.children[0].len();
558                let (child, insertion_index) = if first_child_len < index {
559                    (&mut root.children[1], index - (first_child_len + 1))
560                } else {
561                    (&mut root.children[0], index)
562                };
563                root.length += 1;
564                child.insert_into_non_full_node(insertion_index, element, osd);
565                root.index_insert(element.as_op(osd))
566            } else {
567                root.insert_into_non_full_node(index, element, osd)
568            }
569        } else {
570            let mut root = OpTreeNode::new(self.has_index);
571            root.insert_into_non_full_node(index, element, osd);
572            self.root_node = Some(root)
573        }
574        assert_eq!(self.len(), old_len + 1, "{:#?}", self);
575    }

589    /// Removes the element at `index` from the sequence.
590    ///
591    /// # Panics
592    ///
593    /// Panics if `index` is out of bounds.
594    pub(crate) fn remove(&mut self, index: usize, osd: &OpSetData) -> OpIdx {
595        if let Some(root) = self.root_node.as_mut() {
596            #[cfg(debug_assertions)]
597            let len = root.check();
598            let old = root.remove(index, osd);
599
600            if root.elements.is_empty() {
601                if root.is_leaf() {
602                    self.root_node = None;
603                } else {
604                    self.root_node = Some(root.children.remove(0));
605                }
606            }
607
608            #[cfg(debug_assertions)]
609            debug_assert_eq!(len, self.root_node.as_ref().map_or(0, |r| r.check()) + 1);
610            old
611        } else {
612            panic!("remove from empty tree")
613        }
614    }
