55    /// Parse a document chunk. Input must be the entire chunk including the header and magic
56    /// bytes but the header must already have been parsed. That is to say, this is expected to be
57    /// used like so:
58    ///
59    /// ```rust,ignore
60    /// # use automerge::storage::{parse::{ParseResult, Input}, Document, Header};
61    /// # fn main() -> ParseResult<(), ()> {
62    /// let chunkbytes: &[u8] = todo!();
63    /// let input = Input::new(chunkbytes);
64    /// let (i, header) = Header::parse(input)?;
65    /// let (i, doc) = Document::parse(i, header)?;
66    /// # }
67    /// ```
68    pub(crate) fn parse(
69        input: parse::Input<'a>,
70        header: Header,
71    ) -> parse::ParseResult<'a, Document<'a>, ParseError> {
72        let i = input;
73
74        // Because some columns in a document may be compressed we do some funky stuff when
75        // parsing. As we're parsing the chunk we split the data into four parts:
76        //
77        // .----------------.
78        // | Prefix         |
79        // |.--------------.|
80        // || Actors       ||
81        // || Heads        ||
82        // || Change Meta  ||
83        // || Ops Meta     ||
84        // |'--------------'|
85        // +----------------+
86        // | Change data    |
87        // +----------------+
88        // | Ops data       |
89        // +----------------+
90        // | Suffix         |
91        // |.--------------.|
92        // || Head indices ||
93        // |'--------------'|
94        // '----------------'
95        //
96        // We record the range of each of these sections using `parse::range_of`. Later, we check
97        // if any of the column definitions in change meta or ops meta specify that their columns
98        // are compressed. If there are compressed columns then we copy the uncompressed parts of the
99        // input data to a new output vec, then decompress the compressed parts. Specifically we do
100        // the following:
101        //
102        // * Copy everything in prefix to the output buffer
103        // * If any of change columns are compressed, copy all of change data to the output buffer
104        //   decompressing each compressed column
105        // * Likewise if any of ops columns are compressed copy the data decompressing as required
106        // * Finally copy the suffix
107        //
108        // The reason for all this work is that we end up keeping all of the data behind the
109        // document chunk in a single Vec, which plays nicely with the cache and makes dumping the
110        // document to disk or network straightforward.
111
112        // parse everything in the prefix
113        let (
114            i,
115            parse::RangeOf {
116                range: prefix,
117                value: (actors, heads, change_meta, ops_meta),
118            },
119        ) = parse::range_of(
120            |i| -> parse::ParseResult<'_, _, ParseError> {
121                let (i, actors) = parse::length_prefixed(parse::actor_id)(i)?;
122                let (i, heads) = parse::length_prefixed(parse::change_hash)(i)?;
123                let (i, change_meta) = RawColumns::parse::<ParseError>(i)?;
124                let (i, ops_meta) = RawColumns::parse::<ParseError>(i)?;
125                Ok((i, (actors, heads, change_meta, ops_meta)))
126            },
127            i,
128        )?;
129
130        // parse the change data
131        let (i, parse::RangeOf { range: changes, .. }) =
132            parse::range_of(|i| parse::take_n(change_meta.total_column_len(), i), i)?;
133
134        // parse the ops data
135        let (i, parse::RangeOf { range: ops, .. }) =
136            parse::range_of(|i| parse::take_n(ops_meta.total_column_len(), i), i)?;
137
138        // parse the suffix, which may be empty if this document was produced by an older version
139        // of the JS automerge implementation
140        let (i, suffix, head_indices) = if i.is_empty() {
141            (i, 0..0, Vec::new())
142        } else {
143            let (
144                i,
145                parse::RangeOf {
146                    range: suffix,
147                    value: head_indices,
148                },
149            ) = parse::range_of(
150                |i| parse::apply_n(heads.len(), parse::leb128_u64::<ParseError>)(i),
151                i,
152            )?;
153            (i, suffix, head_indices)
154        };
155
156        let compression::Decompressed {
157            change_bytes,
158            op_bytes,
159            uncompressed,
160            compressed,
161            changes,
162            ops,
163        } = compression::decompress(compression::Args {
164            prefix: prefix.start,
165            suffix: suffix.start,
166            original: Cow::Borrowed(input.bytes()),
167            changes: compression::Cols {
168                data: changes,
169                raw_columns: change_meta,
170            },
171            ops: compression::Cols {
172                data: ops,
173                raw_columns: ops_meta,
174            },
175            extra_args: (),
176        })
177        .map_err(|e| parse::ParseError::Error(ParseError::RawColumns(e)))?;
178
179        let ops_layout = Columns::parse(op_bytes.len(), ops.iter()).map_err(|e| {
180            parse::ParseError::Error(ParseError::BadColumnLayout {
181                column_type: "ops",
182                error: e,
183            })
184        })?;
185        let ops_cols =
186            DocOpColumns::try_from(ops_layout).map_err(|e| parse::ParseError::Error(e.into()))?;
187
188        let change_layout = Columns::parse(change_bytes.len(), changes.iter()).map_err(|e| {
189            parse::ParseError::Error(ParseError::BadColumnLayout {
190                column_type: "changes",
191                error: e,
192            })
193        })?;
194        let change_cols = DocChangeColumns::try_from(change_layout)
195            .map_err(|e| parse::ParseError::Error(e.into()))?;
196
197        Ok((
198            i,
199            Document {
200                bytes: uncompressed,
201                compressed_bytes: compressed,
202                header,
203                actors,
204                heads,
205                op_metadata: ops_cols,
206                op_bytes,
207                change_metadata: change_cols,
208                change_bytes,
209                head_indices,
210            },
211        ))
212    }

214    pub(crate) fn new<'b, I, C, IC, D, O>(
215        mut actors: Vec<ActorId>,
216        heads_with_indices: Vec<(ChangeHash, usize)>,
217        ops: I,
218        changes: IC,
219        compress: CompressConfig,
220    ) -> Document<'static>
221    where
222        I: Iterator<Item = D> + Clone + ExactSizeIterator,
223        O: convert::OpId<usize>,
224        D: AsDocOp<'b, OpId = O>,
225        C: AsChangeMeta<'b>,
226        IC: Iterator<Item = C> + Clone,
227    {
228        let mut ops_out = Vec::new();
229        let ops_meta = DocOpColumns::encode(ops, &mut ops_out);
230
231        let mut change_out = Vec::new();
232        let change_meta = DocChangeColumns::encode(changes, &mut change_out);
233        actors.sort_unstable();
234
235        let mut data = Vec::with_capacity(ops_out.len() + change_out.len());
236        leb128::write::unsigned(&mut data, actors.len() as u64).unwrap();
237        for actor in &actors {
238            leb128::write::unsigned(&mut data, actor.to_bytes().len() as u64).unwrap();
239            data.extend(actor.to_bytes());
240        }
241        leb128::write::unsigned(&mut data, heads_with_indices.len() as u64).unwrap();
242        for (head, _) in &heads_with_indices {
243            data.extend(head.as_bytes());
244        }
245        let prefix_len = data.len();
246
247        change_meta.raw_columns().write(&mut data);
248        ops_meta.raw_columns().write(&mut data);
249        let change_start = data.len();
250        let change_end = change_start + change_out.len();
251        data.extend(change_out);
252        let ops_start = data.len();
253        let ops_end = ops_start + ops_out.len();
254        data.extend(ops_out);
255        let suffix_start = data.len();
256
257        let head_indices = heads_with_indices
258            .iter()
259            .map(|(_, i)| *i as u64)
260            .collect::<Vec<_>>();
261        for index in &head_indices {
262            leb128::write::unsigned(&mut data, *index).unwrap();
263        }
264
265        let header = Header::new(ChunkType::Document, &data);
266        let mut bytes = Vec::with_capacity(data.len() + header.len());
267        header.write(&mut bytes);
268        let header_len = bytes.len();
269        bytes.extend(&data);
270
271        let op_bytes = shift_range(ops_start..ops_end, header.len());
272        let change_bytes = shift_range(change_start..change_end, header.len());
273
274        let compressed_bytes = if let CompressConfig::Threshold(threshold) = compress {
275            let compressed = Cow::Owned(compression::compress(compression::Args {
276                prefix: prefix_len + header.len(),
277                suffix: suffix_start + header.len(),
278                ops: compression::Cols {
279                    raw_columns: ops_meta.raw_columns(),
280                    data: op_bytes.clone(),
281                },
282                changes: compression::Cols {
283                    raw_columns: change_meta.raw_columns(),
284                    data: change_bytes.clone(),
285                },
286                original: Cow::Borrowed(&bytes),
287                extra_args: compression::CompressArgs {
288                    threshold,
289                    original_header_len: header_len,
290                },
291            }));
292            Some(compressed)
293        } else {
294            None
295        };
296
297        Document {
298            actors,
299            bytes: Cow::Owned(bytes),
300            compressed_bytes,
301            header,
302            heads: heads_with_indices.into_iter().map(|(h, _)| h).collect(),
303            op_metadata: ops_meta,
304            op_bytes,
305            change_metadata: change_meta,
306            change_bytes,
307            head_indices,
308        }
309    }

311    pub(crate) fn iter_ops(
312        &'a self,
313    ) -> impl Iterator<Item = Result<DocOp, ReadDocOpError>> + Clone + 'a {
314        self.op_metadata.iter(&self.bytes[self.op_bytes.clone()])
315    }
316
317    pub(crate) fn iter_changes(
318        &'a self,
319    ) -> impl Iterator<Item = Result<ChangeMetadata<'_>, ReadChangeError>> + Clone + 'a {
320        self.change_metadata
321            .iter(&self.bytes[self.change_bytes.clone()])
322    }

324    pub(crate) fn into_bytes(self) -> Vec<u8> {
325        if let Some(compressed) = self.compressed_bytes {
326            compressed.into_owned()
327        } else {
328            self.bytes.into_owned()
329        }
330    }
