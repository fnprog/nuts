37impl<T: Clone + Decodable + Encodable + PartialEq + Eq + Debug> RleRange<T> {
38    /// The semantics of this are similar to `Vec::splice`
39    ///
40    /// # Arguments
41    ///
42    /// * `data` - The buffer containing the original rows
43    /// * `replace` - The range of elements in the original collection to replace
44    /// * `replace_with` - An iterator to insert in place of the original elements.
45    /// * `out` - The buffer to encode the resulting collection into
46    pub(crate) fn splice<
47        'a,
48        I: Iterator<Item = Result<Option<TB>, E>>,
49        TB: Borrow<T> + 'a,
50        E: std::error::Error,
51    >(
52        &self,
53        data: &[u8],
54        replace: Range<usize>,
55        mut replace_with: I,
56        out: &mut Vec<u8>,
57    ) -> Result<Self, SpliceError<raw::Error, E>> {
58        let start = out.len();
59        let mut encoder = self.encoder(out);
60        let mut decoder = self.decoder(data);
61        let mut idx = 0;
62        while idx < replace.start {
63            match decoder
64                .next()
65                .transpose()
66                .map_err(SpliceError::ReadExisting)?
67            {
68                Some(elem) => encoder.append(elem.as_ref()),
69                None => panic!("out of bounds"),
70            }
71            idx += 1;
72        }
73        for _ in 0..replace.len() {
74            decoder.next();
75            if let Some(next) = replace_with
76                .next()
77                .transpose()
78                .map_err(SpliceError::ReadReplace)?
79            {
80                encoder.append(next.as_ref().map(|n| n.borrow()));
81            }
82        }
83        for next in replace_with {
84            let next = next.map_err(SpliceError::ReadReplace)?;
85            encoder.append(next.as_ref().map(|n| n.borrow()));
86        }
87        for next in decoder {
88            let next = next.map_err(SpliceError::ReadExisting)?;
89            encoder.append(next.as_ref());
90        }
91        let (_, len) = encoder.finish();
92        let range = start..(start + len);
93        Ok(range.into())
94    }
95}

97impl<'a, T: Encodable + Clone + PartialEq + 'a> RleRange<T> {
98    pub(crate) fn encoder<S: Sink>(&self, output: S) -> RleEncoder<S, T> {
99        RleEncoder::from(output)
100    }
101
102    pub(crate) fn encode<BT: Borrow<T>, I: Iterator<Item = Option<BT>>>(
103        items: I,
104        out: &mut Vec<u8>,
105    ) -> Self {
106        let start = out.len();
107        let mut encoder = RleEncoder::new(out);
108        for item in items {
109            encoder.append(item);
110        }
111        let (_, len) = encoder.finish();
112        (start..(start + len)).into()
113    }
114}

145    fn rle_int_round_trip() {
146        let vals = [1, 1, 2, 2, 3, 2, 3, 1, 3];
147        let mut buf = Vec::with_capacity(vals.len() * 3);
148        let mut encoder: RleEncoder<_, u64> = RleEncoder::new(&mut buf);
149        for val in vals {
150            encoder.append_value(val)
151        }
152        let (_, total_slice_len) = encoder.finish();
153        let mut decoder: RleDecoder<'_, u64> =
154            RleDecoder::from(Cow::Borrowed(&buf[0..total_slice_len]));
155        let mut result = Vec::new();
156        while let Some(Some(val)) = decoder.next().transpose().unwrap() {
157            result.push(val);
158        }
159        assert_eq!(result, vals);
160    }

162    #[test]
163    fn rle_int_insert() {
164        let vals = [1, 1, 2, 2, 3, 2, 3, 1, 3];
165        let mut buf = Vec::with_capacity(vals.len() * 3);
166        let mut encoder: RleEncoder<_, u64> = RleEncoder::new(&mut buf);
167        for val in vals.iter().take(4) {
168            encoder.append_value(val)
169        }
170        encoder.append_value(5);
171        for val in vals.iter().skip(4) {
172            encoder.append_value(val);
173        }
174        let (_, total_slice_len) = encoder.finish();
175        let mut decoder: RleDecoder<'_, u64> =
176            RleDecoder::from(Cow::Borrowed(&buf[0..total_slice_len]));
177        let mut result = Vec::new();
178        while let Some(Some(val)) = decoder.next().transpose().unwrap() {
179            result.push(val);
180        }
181        let expected = [1, 1, 2, 2, 5, 3, 2, 3, 1, 3];
182        assert_eq!(result, expected);
183    }

185    fn encode<T: Clone + Encodable + PartialEq>(vals: &[Option<T>]) -> (RleRange<T>, Vec<u8>) {
186        let mut buf = Vec::with_capacity(vals.len() * 3);
187        let range = RleRange::<T>::encode(vals.iter().map(|v| v.as_ref()), &mut buf);
188        (range, buf)
189    }
190
191    fn decode<T: Clone + Decodable + Debug>(range: RleRange<T>, buf: &[u8]) -> Vec<Option<T>> {
192        range.decoder(buf).collect::<Result<Vec<_>, _>>().unwrap()
193    }
