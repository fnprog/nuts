8/// The graph of changes
9///
10/// This is a sort of adjacency list based representation, except that instead of using linked
11/// lists, we keep all the edges and nodes in two vecs and reference them by index which plays nice
12/// with the cache
13#[derive(Debug, Clone)]
14pub(crate) struct ChangeGraph {
15    nodes: Vec<ChangeNode>,
16    edges: Vec<Edge>,
17    hashes: Vec<ChangeHash>,
18    nodes_by_hash: BTreeMap<ChangeHash, NodeIdx>,
19    clock_cache: Vec<Clock>,
20}
21
22const CACHE_STEP: u32 = 32;
23
24#[derive(Clone, Copy, Debug, PartialEq, Eq, PartialOrd, Ord)]
25struct NodeIdx(u32);
26
27#[derive(Clone, Copy, Debug, PartialEq, Eq, PartialOrd, Ord)]
28struct EdgeIdx(u32);
29
30#[derive(Clone, Copy, Debug, PartialEq, Eq, PartialOrd, Ord)]
31struct HashIdx(u32);
32
33#[derive(Debug, Clone)]
34struct Edge {
35    // Edges are always child -> parent so we only store the target, the child is implicit
36    // as you get the edge from the child
37    target: NodeIdx,
38    next: Option<EdgeIdx>,
39}
40
41#[derive(Debug, Clone)]
42struct ChangeNode {
43    hash_idx: HashIdx,
44    actor_index: usize,
45    seq: u64,
46    max_op: u64,
47    parents: Option<EdgeIdx>,
48}
49
50impl ChangeGraph {
51    pub(crate) fn new() -> Self {
52        Self {
53            nodes: Vec::new(),
54            edges: Vec::new(),
55            nodes_by_hash: BTreeMap::new(),
56            hashes: Vec::new(),
57            clock_cache: Vec::new(),
58        }
59    }
60
61    pub(crate) fn add_change(
62        &mut self,
63        change: &Change,
64        actor_idx: usize,
65    ) -> Result<(), MissingDep> {
66        let hash = change.hash();
67        if self.nodes_by_hash.contains_key(&hash) {
68            return Ok(());
69        }
70        let parent_indices = change
71            .deps()
72            .iter()
73            .map(|h| self.nodes_by_hash.get(h).copied().ok_or(MissingDep(*h)))
74            .collect::<Result<Vec<_>, _>>()?;
75        let node_idx = self.add_node(actor_idx, change);
76        self.nodes_by_hash.insert(hash, node_idx);
77        for parent_idx in parent_indices {
78            self.add_parent(node_idx, parent_idx);
79        }
80        if let Some(cached_idx) = Self::node_to_cache(&node_idx, CACHE_STEP) {
81            assert_eq!(cached_idx, self.clock_cache.len());
82            let clock = self.calculate_clock(vec![node_idx]);
83            self.clock_cache.push(clock)
84        }
85        Ok(())
86    }
87
88    fn add_node(&mut self, actor_index: usize, change: &Change) -> NodeIdx {
89        let idx = NodeIdx(self.nodes.len() as u32);
90        let hash_idx = self.add_hash(change.hash());
91        self.nodes.push(ChangeNode {
92            hash_idx,
93            actor_index,
94            seq: change.seq(),
95            max_op: change.max_op(),
96            parents: None,
97        });
98        idx
99    }
100
101    fn add_hash(&mut self, hash: ChangeHash) -> HashIdx {
102        let idx = HashIdx(self.hashes.len() as u32);
103        self.hashes.push(hash);
104        idx
105    }
106
107    fn add_parent(&mut self, child_idx: NodeIdx, parent_idx: NodeIdx) {
108        let new_edge_idx = EdgeIdx(self.edges.len() as u32);
109        let new_edge = Edge {
110            target: parent_idx,
111            next: None,
112        };
113        self.edges.push(new_edge);
114
115        let child = &mut self.nodes[child_idx.0 as usize];
116        if let Some(edge_idx) = child.parents {
117            let mut edge = &mut self.edges[edge_idx.0 as usize];
118            while let Some(next) = edge.next {
119                edge = &mut self.edges[next.0 as usize];
120            }
121            edge.next = Some(new_edge_idx);
122        } else {
123            child.parents = Some(new_edge_idx);
124        }
125    }
126
127    fn parents(&self, node_idx: NodeIdx) -> impl Iterator<Item = NodeIdx> + '_ {
128        let mut edge_idx = self.nodes[node_idx.0 as usize].parents;
129        std::iter::from_fn(move || {
130            let this_edge_idx = edge_idx?;
131            let edge = &self.edges[this_edge_idx.0 as usize];
132            edge_idx = edge.next;
133            Some(edge.target)
134        })
135    }
136
137    fn heads_to_nodes(&self, heads: &[ChangeHash]) -> Vec<NodeIdx> {
138        heads
139            .iter()
140            .filter_map(|h| self.nodes_by_hash.get(h))
141            .copied()
142            .collect()
143    }
144
145    pub(crate) fn clock_for_heads(&self, heads: &[ChangeHash]) -> Clock {
146        let nodes = self.heads_to_nodes(heads);
147        assert_eq!(
148            self.clock_cache.len(),
149            self.nodes.len() / CACHE_STEP as usize
150        );
151        self.calculate_clock(nodes)
152    }
153
154    fn node_to_cache(idx: &NodeIdx, step: u32) -> Option<usize> {
155        assert!(step > 2);
156        if (idx.0 + 1) % step == 0 {
157            Some(((idx.0 + 1) / step - 1) as usize)
158        } else {
159            None
160        }
161    }
162
163    fn calculate_clock(&self, nodes: Vec<NodeIdx>) -> Clock {
164        let mut clock = Clock::new();
165
166        self.traverse_ancestors(nodes, |node, idx| {
167            clock.include(
168                node.actor_index,
169                ClockData {
170                    max_op: node.max_op,
171                    seq: node.seq,
172                },
173            );
174            if let Some(cached_idx) = Self::node_to_cache(&idx, CACHE_STEP) {
175                if cached_idx < self.clock_cache.len() {
176                    let ancestor_clock = &self.clock_cache[cached_idx];
177                    clock = Clock::merge(&clock, ancestor_clock);
178                    return false; // dont look at ancestors
179                }
180            }
181            true // do look at ancestors
182        });
183
184        clock
185    }
186
187    pub(crate) fn remove_ancestors(
188        &self,
189        changes: &mut BTreeSet<ChangeHash>,
190        heads: &[ChangeHash],
191    ) {
192        let nodes = self.heads_to_nodes(heads);
193        self.traverse_ancestors(nodes, |node, _idx| {
194            let hash = &self.hashes[node.hash_idx.0 as usize];
195            changes.remove(hash);
196            true
197        });
198    }
199
200    /// Call `f` for each (node, hash) in the graph, starting from the given heads
201    ///
202    /// No guarantees are made about the order of traversal but each node will only be visited
203    /// once.
204    fn traverse_ancestors<F: FnMut(&ChangeNode, NodeIdx) -> bool>(
205        &self,
206        mut to_visit: Vec<NodeIdx>,
207        mut f: F,
208    ) {
209        let mut visited = BTreeSet::new();
210
211        while let Some(idx) = to_visit.pop() {
212            if visited.contains(&idx) {
213                continue;
214            } else {
215                visited.insert(idx);
216            }
217            let node = &self.nodes[idx.0 as usize];
218            if f(node, idx) {
219                to_visit.extend(self.parents(idx));
220            }
221        }
222    }
223}

290    struct TestGraphBuilder {
291        actors: Vec<ActorId>,
292        changes: Vec<Change>,
293        seqs_by_actor: BTreeMap<ActorId, u64>,
294    }
295
296    impl TestGraphBuilder {
297        fn new() -> Self {
298            TestGraphBuilder {
299                actors: Vec::new(),
300                changes: Vec::new(),
301                seqs_by_actor: BTreeMap::new(),
302            }
303        }
304
305        fn actor(&mut self) -> ActorId {
306            let actor = ActorId::random();
307            self.actors.push(actor.clone());
308            actor
309        }
310
311        fn index(&self, actor: &ActorId) -> usize {
312            self.actors.iter().position(|a| a == actor).unwrap()
313        }
314
315        /// Create a change with `num_new_ops` and `parents` for `actor`
316        ///
317        /// The `start_op` and `seq` of the change will be computed from the
318        /// previous changes for the same actor.
319        fn change(
320            &mut self,
321            actor: &ActorId,
322            num_new_ops: usize,
323            parents: &[ChangeHash],
324        ) -> ChangeHash {
325            let mut osd = OpSetData::from_actors(self.actors.clone());
326            let key = osd.props.cache("key".to_string());
327
328            let start_op = parents
329                .iter()
330                .map(|c| {
331                    self.changes
332                        .iter()
333                        .find(|change| change.hash() == *c)
334                        .unwrap()
335                        .max_op()
336                })
337                .max()
338                .unwrap_or(0)
339                + 1;
340
341            let actor_idx = self.index(actor);
342            let ops = (0..num_new_ops)
343                .map(|opnum| OpBuilder {
344                    id: OpId::new(start_op + opnum as u64, actor_idx),
345                    action: crate::OpType::Put("value".into()),
346                    key: Key::Map(key),
347                    insert: false,
348                })
349                .collect::<Vec<_>>();
350
351            let root = ObjId::root();
352            let timestamp = SystemTime::now()
353                .duration_since(UNIX_EPOCH)
354                .unwrap()
355                .as_millis() as i64;
356            let seq = self.seqs_by_actor.entry(actor.clone()).or_insert(1);
357            let ops = ops
358                .into_iter()
359                .map(|op| osd.push(root, op))
360                .collect::<Vec<_>>();
361            let change = Change::new(
362                ChangeBuilder::new()
363                    .with_dependencies(parents.to_vec())
364                    .with_start_op(NonZeroU64::new(start_op).unwrap())
365                    .with_actor(actor.clone())
366                    .with_seq(*seq)
367                    .with_timestamp(timestamp)
368                    .build(ops.iter().map(|op| op_as_actor_id(op.as_op(&osd))))
369                    .unwrap(),
370            );
371            *seq = seq.checked_add(1).unwrap();
372            let hash = change.hash();
373            self.changes.push(change);
374            hash
375        }
376
377        fn build(&self) -> ChangeGraph {
378            let mut graph = ChangeGraph::new();
379            for change in &self.changes {
380                let actor_idx = self.index(change.actor_id());
381                graph.add_change(change, actor_idx).unwrap();
382            }
383            graph
384        }
385    }
