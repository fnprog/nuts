10impl<'a> OpTreeIter<'a> {
11    pub(crate) fn new(tree: &'a OpTreeInternal) -> OpTreeIter<'a> {
12        Self(
13            tree.root_node
14                .as_ref()
15                .map(|root| Inner::NonEmpty {
16                    // This is a guess at the average depth of an OpTree
17                    ancestors: Vec::with_capacity(6),
18                    current: NodeIter {
19                        node: root,
20                        index: 0,
21                    },
22                    cumulative_index: 0,
23                    root_node: root,
24                })
25                .unwrap_or(Inner::Empty),
26        )
27    }
28}

42#[derive(Clone)]
43enum Inner<'a> {
44    Empty,
45    NonEmpty {
46        // A stack of nodes in the optree which we have descended in to to get to the current
47        // element.
48        ancestors: Vec<NodeIter<'a>>,
49        current: NodeIter<'a>,
50        // How far through the whole optree we are
51        cumulative_index: usize,
52        root_node: &'a OpTreeNode,
53    },
54}
55
56/// A node in the op tree which we are iterating over
57#[derive(Clone)]
58struct NodeIter<'a> {
59    /// The node itself
60    node: &'a OpTreeNode,
61    /// The index of the next element we will pull from the node. This means something different
62    /// depending on whether the node is a leaf node or not. If the node is a leaf node then this
63    /// index is the index in `node.elements` which will be returned on the next call to `next()`.
64    /// If the node is not an internal node then this index is the index of `children` which we are
65    /// currently iterating as well as being the index of the next element of `elements` which we
66    /// will return once we have finished iterating over the child node.
67    index: usize,
68}
69
70impl<'a> Iterator for Inner<'a> {
71    type Item = OpIdx;
72
73    fn next(&mut self) -> Option<Self::Item> {
74        match self {
75            Inner::Empty => None,
76            Inner::NonEmpty {
77                ancestors,
78                current,
79                cumulative_index,
80                ..
81            } => {
82                if current.node.is_leaf() {
83                    // If we're in a leaf node and we haven't exhausted it yet we just return the elements
84                    // of the leaf node
85                    if current.index < current.node.len() {
86                        let result = current.node.elements[current.index];
87                        current.index += 1;
88                        *cumulative_index += 1;
89                        Some(result)
90                    } else {
91                        // We've exhausted the leaf node, we must find the nearest non-exhausted parent (lol)
92                        let node_iter = loop {
93                            if let Some(
94                                node_iter @ NodeIter {
95                                    node: parent,
96                                    index: parent_index,
97                                },
98                            ) = ancestors.pop()
99                            {
100                                // We've exhausted this parent
101                                if parent_index >= parent.elements.len() {
102                                    continue;
103                                } else {
104                                    // This parent still has elements to process, let's use it!
105                                    break node_iter;
106                                }
107                            } else {
108                                // No parents left, we're done
109                                return None;
110                            }
111                        };
112                        // if we've finished the elements in a leaf node and there's a parent node then we
113                        // return the element from the parent node which is one after the index at which we
114                        // descended into the child
115                        *current = node_iter;
116                        let result = current.node.elements[current.index];
117                        current.index += 1;
118                        *cumulative_index += 1;
119                        Some(result)
120                    }
121                } else {
122                    // If we're in a non-leaf node then the last iteration returned an element from the
123                    // current nodes `elements`, so we must now descend into a leaf child
124                    ancestors.push(current.clone());
125                    loop {
126                        let child = &current.node.children[current.index];
127                        current.index = 0;
128                        if !child.is_leaf() {
129                            ancestors.push(NodeIter {
130                                node: child,
131                                index: 0,
132                            });
133                            current.node = child
134                        } else {
135                            current.node = child;
136                            break;
137                        }
138                    }
139                    self.next()
140                }
141            }
142        }
143    }
144
145    fn nth(&mut self, n: usize) -> Option<Self::Item> {
146        match self {
147            Self::Empty => None,
148            Self::NonEmpty {
149                root_node,
150                cumulative_index,
151                current,
152                ancestors,
153                ..
154            } => {
155                // Make sure that we don't rewind when calling nth more than once
156                if n < *cumulative_index {
157                    None
158                } else if n >= root_node.len() {
159                    *cumulative_index = root_node.len() - 1;
160                    None
161                } else {
162                    // rather than trying to go back up through the ancestors to find the right
163                    // node we just start at the root.
164                    *current = NodeIter {
165                        node: root_node,
166                        index: n,
167                    };
168                    *cumulative_index = 0;
169                    ancestors.clear();
170                    while !current.node.is_leaf() {
171                        for (child_index, child) in current.node.children.iter().enumerate() {
172                            match (*cumulative_index + child.len()).cmp(&n) {
173                                Ordering::Less => {
174                                    *cumulative_index += child.len() + 1;
175                                    current.index = child_index + 1;
176                                }
177                                Ordering::Equal => {
178                                    *cumulative_index += child.len() + 1;
179                                    current.index = child_index + 1;
180                                    return Some(current.node.elements[child_index]);
181                                }
182                                Ordering::Greater => {
183                                    current.index = child_index;
184                                    let old = std::mem::replace(
185                                        current,
186                                        NodeIter {
187                                            node: child,
188                                            index: 0,
189                                        },
190                                    );
191                                    ancestors.push(old);
192                                    break;
193                                }
194                            }
195                        }
196                    }
197                    // we're in a leaf node and we kept track of the cumulative index as we went,
198                    let index_in_this_node = n.saturating_sub(*cumulative_index);
199                    current.index = index_in_this_node + 1;
200                    Some(current.node.elements[index_in_this_node])
201                }
202            }
203        }
204    }
205}

248    // A struct which impls Debug by only printing the counters of the IDs of the  ops it wraps.
249    // This is useful because the only difference between the ops that we generate is the counter
250    // of their IDs. Wrapping a Vec<OpBuilder> in DebugOps will result in output from assert! etc. which
251    // only shows the counters. For example, the output of a failing assert_eq! like this
252    //
253    //     assert_eq!(DebugOps(&ops1), DebugOps(&ops2))
254    //
255    // Might look like this
256    //
257    //     left: `[0,1,2,3]
258    //     right: `[0,1,2,3,4]
259    //
260    // i.e. all the other details of the ops are elided
261    #[derive(PartialEq)]
262    struct DebugOps<'a>(&'a [OpIdx]);
263
264    impl<'a> std::fmt::Debug for DebugOps<'a> {
265        fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
266            write!(f, "[")?;
267            for (index, op) in self.0.iter().enumerate() {
268                if index < self.0.len() - 1 {
269                    write!(f, "{:?},", op)?;
270                } else {
271                    write!(f, "{:?}]", op)?
272                }
273            }
274            Ok(())
275        }
276    }

278    fn op(counter: u64, osd: &mut OpSetData) -> OpIdx {
279        let op = OpBuilder {
280            action: OpType::Put(ScalarValue::Uint(counter)),
281            id: OpId::new(counter, 0),
282            key: Key::Map(0),
283            insert: false,
284        };
285        osd.push(ROOT.into(), op)
286    }

288    /// A model for a property based test of the OpTreeIter. We generate a set of actions, each
289    /// action pertaining to a `model` - which is just a `Vec<OpBuilder>`. As we generate each action we
290    /// apply it to the model and record the action we took. In the property test we replay the
291    /// same actions against an `OpTree` and check that the iterator returns the same result as the
292    /// `model`.
293    #[derive(Clone)]
294    struct Model {
295        actions: Vec<Action>,
296        model: Vec<OpIdx>,
297        osd: OpSetData,
298    }
299
300    impl std::fmt::Debug for Model {
301        fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
302            f.debug_struct("Model")
303                .field("actions", &self.actions)
304                .field("model", &DebugOps(&self.model))
305                .finish()
306        }
307    }

309    impl Model {
310        fn insert(&self, index: usize, next_op_counter: u64) -> Self {
311            let mut actions = self.actions.clone();
312            let mut osd = OpSetData::default();
313            let op = op(next_op_counter, &mut osd);
314            actions.push(Action::Insert(index, op));
315            let mut model = self.model.clone();
316            model.insert(index, op);
317            Self {
318                actions,
319                model,
320                osd,
321            }
322        }
323
324        fn delete(&self, index: usize) -> Self {
325            let mut actions = self.actions.clone();
326            actions.push(Action::Delete(index));
327            let mut model = self.model.clone();
328            let osd = self.osd.clone();
329            model.remove(index);
330            Self {
331                actions,
332                model,
333                osd,
334            }
335        }
336
337        fn next(self, next_op_counter: u64) -> impl Strategy<Value = Model> {
338            if self.model.is_empty() {
339                Just(self.insert(0, next_op_counter)).boxed()
340            } else {
341                // Note that we have to feed `self` through the `prop_flat_map` using `Just` to
342                // appease the borrow checker, this is annoying because it does obscure the meaning
343                // of the code heere which is basically "decide whether the next action should be
344                // insert, if it is insert choose an index between 0..model.len() + 1 and generate
345                // an op to insert, otherwise choose an index between 0..model.len() and generate a
346                // delete action".
347                //
348                // 95% chance of inserting to make sure we deal with large lists
349                (proptest::bool::weighted(0.95), Just(self))
350                    .prop_flat_map(move |(insert, model)| {
351                        if insert {
352                            (0..model.model.len() + 1, Just(model))
353                                .prop_map(move |(index, model)| {
354                                    model.insert(index, next_op_counter)
355                                })
356                                .boxed()
357                        } else {
358                            ((0..model.model.len()), Just(model))
359                                .prop_map(move |(index, model)| model.delete(index))
360                                .boxed()
361                        }
362                    })
363                    .boxed()
364            }
365        }
366    }
367
368    fn model() -> impl Strategy<Value = Model> {
369        (0_u64..150).prop_flat_map(|num_steps| {
370            let mut strat = Just((
371                0,
372                Model {
373                    actions: Vec::new(),
374                    model: Vec::new(),
375                    osd: OpSetData::default(),
376                },
377            ))
378            .boxed();
379            for _ in 0..num_steps {
380                strat = strat
381                    // Note the counter, which we feed through each `prop_flat_map`, incrementing
382                    // it by one each time. This mean that the generated ops have ascending (but
383                    // not necessarily consecutive because not every `Action` is an `Insert`)
384                    // counters. This makes it easier to debug failures - if we just used a random
385                    // counter it would be much harder to see where things are out of order.
386                    .prop_flat_map(|(counter, model)| {
387                        let next_counter = counter + 1;
388                        model.next(counter).prop_map(move |m| (next_counter, m))
389                    })
390                    .boxed();
391            }
392            strat.prop_map(|(_, model)| model)
393        })
394    }

396    fn make_optree(actions: &[Action], osd: &OpSetData) -> super::OpTreeInternal {
397        let mut optree = OpTreeInternal::new(ObjType::List);
398        for action in actions {
399            match action {
400                Action::Insert(index, idx) => {
401                    optree.insert(*index, *idx, osd);
402                }
403                Action::Delete(index) => {
404                    optree.remove(*index, osd);
405                }
406            }
407        }
408        optree
409    }

430    fn nth_model() -> impl Strategy<Value = NthModel> {
431        model().prop_flat_map(|model| {
432            if model.model.is_empty() {
433                Just(NthModel {
434                    model: model.model,
435                    actions: model.actions,
436                    osd: model.osd,
437                    n: 0,
438                })
439                .boxed()
440            } else {
441                (0..model.model.len(), Just(model))
442                    .prop_map(|(index, model)| NthModel {
443                        model: model.model,
444                        actions: model.actions,
445                        osd: model.osd,
446                        n: index,
447                    })

449            }

455        fn optree_iter_proptest(model in model()) {
456            let optree = make_optree(&model.actions, &model.osd);
457            let iter = super::OpTreeIter::new(&optree);

466            let mut model_iter = model.model.iter();
