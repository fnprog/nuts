21/// An automerge document that automatically manages transactions.
22///
23/// ## Creating, loading, merging and forking documents
24///
25/// A new document can be created with [`Self::new()`], which will create a document with a random
26/// [`ActorId`]. Existing documents can be loaded with [`Self::load()`].
27///
28/// If you have two documents and you want to merge the changes from one into the other you can use
29/// [`Self::merge()`].
30///
31/// If you have a document you want to split into two concurrent threads of execution you can use
32/// [`Self::fork()`]. If you want to split a document from ealier in its history you can use
33/// [`Self::fork_at()`].
34///
35/// ## Reading values
36///
37/// [`Self`] implements [`ReadDoc`], which provides methods for reading values from the document.
38///
39/// ## Modifying a document
40///
41/// This type implements [`Transactable`] directly, so you can modify it using methods from [`Transactable`].
42///
43/// ## Synchronization
44///
45/// To synchronise call [`Self::sync()`] which returns an implementation of [`SyncDoc`]
46///
47/// ## Patches, maintaining materialized views
48///
49/// [`AutoCommit`] allows you to generate [`Patch`]es representing changes to the current state of
50/// the document which you can use to maintain a materialized view of the current state. There are
51/// several ways to use this. See the documentation on [`Self::diff()`] for more details, but the key
52/// point to remember is that [`AutoCommit`] manages an internal "diff cursor" for you. This is a
53/// representation of the heads of the document last time you called [`Self::diff_incremental()`]
54/// but you can also manage it directly using [`Self::update_diff_cursor()`] and
55/// [`Self::reset_diff_cursor()`].
56#[derive(Debug, Clone)]
57pub struct AutoCommit {
58    pub(crate) doc: Automerge,
59    transaction: Option<(PatchLog, TransactionInner)>,
60    patch_log: PatchLog,
61    diff_cursor: Vec<ChangeHash>,
62    diff_cache: Option<(OpRange, Vec<Patch>)>,
63    save_cursor: Vec<ChangeHash>,
64    isolation: Option<Vec<ChangeHash>>,
65}

178    /// Generates a diff from `before` to `after`
179    ///
180    /// By default the diff requires a sequental scan of all the ops in the doc.
181    ///
182    /// To do a fast indexed diff `before` must equal [`Self::diff_cursor()`] and
183    /// `after` must equal [`Self::get_heads()`]. The diff cursor is managed with
184    /// [`Self::update_diff_cursor()`] and [`Self::reset_diff_cursor()`]
185    ///
186    /// Managing the diff index has a small but non-zero overhead.  It should be
187    /// disabled if no longer needed.  If a signifigantly large change is applied
188    /// to the document it may be faster to reset the index before applying it,
189    /// doing an unindxed diff afterwards and then reenable the index.
190    ///
191    /// # Arguments
192    ///
193    /// * `before` - heads from [`Self::get_heads()`] at beginning point in the documents history
194    /// * `after` - heads from [`Self::get_heads()`] at ending point in the documents history.
195    ///
196    /// Note: `before` and `after` do not have to be chronological.  Document state can move backward.
197    /// Normal use might look like:
198    ///
199    /// # Example
200    ///
201    /// ```
202    /// use automerge::{ AutoCommit };
203    ///
204    /// let mut doc = AutoCommit::new(); // or AutoCommit::load(data)
205    /// // make some changes - use and update the index
206    /// let heads = doc.get_heads();
207    /// let diff_cursor = doc.diff_cursor();
208    /// let patches = doc.diff(&diff_cursor, &heads);
209    /// doc.update_diff_cursor();
210    /// ```
211    ///
212    /// See [`Self::diff_incremental()`] for encapsulating this pattern.
213    pub fn diff(&mut self, before: &[ChangeHash], after: &[ChangeHash]) -> Vec<Patch> {
214        self.ensure_transaction_closed();
215        let range = OpRange::new(before, after);
216        if let Some((r, patches)) = &self.diff_cache {
217            if r == &range {
218                // we could skip this clone and return &[Patch]
219                return patches.clone();
220            }
221        }
222        let heads = self.doc.get_heads();
223        let patches = if range.after() == heads
224            && range.before() == self.diff_cursor
225            && self.patch_log.is_active()
226        {
227            self.patch_log.make_patches(&self.doc)
228        } else if range.before().is_empty() && range.after() == heads {
229            let mut patch_log = PatchLog::active(self.patch_log.text_rep());
230            // This if statement is only active if the current heads are the same as `after`
231            // so we don't need to tell the patch log to target a specific heads and consequently
232            // it wll be able to generate patches very fast as it doesn't need to make any clocks
233            patch_log.heads = None;
234            current_state::log_current_state_patches(&self.doc, &mut patch_log);
235            patch_log.make_patches(&self.doc)
236        } else {
237            let before_clock = self.doc.clock_at(range.before());
238            let after_clock = self.doc.clock_at(range.after());
239            let mut patch_log = PatchLog::active(self.patch_log.text_rep());
240            patch_log.heads = Some(range.after().to_vec());
241            diff::log_diff(&self.doc, &before_clock, &after_clock, &mut patch_log);
242            patch_log.make_patches(&self.doc)
243        };
244        self.diff_cache = Some((range, patches));
245        self.diff_cache.as_ref().unwrap().1.clone()
246    }

248    /// This is a convience function that encapsulates the following common pattern
249    /// ```
250    /// use automerge::AutoCommit;
251    /// let mut doc = AutoCommit::new();
252    /// // make some changes
253    /// let heads = doc.get_heads();
254    /// let diff_cursor = doc.diff_cursor();
255    /// let patches = doc.diff(&diff_cursor, &heads);
256    /// doc.update_diff_cursor();
257    /// ```
258    pub fn diff_incremental(&mut self) -> Vec<Patch> {
259        self.ensure_transaction_closed();
260        let heads = self.doc.get_heads();
261        let diff_cursor = self.diff_cursor();
262        let patches = self.diff(&diff_cursor, &heads);
263        self.update_diff_cursor();
264        patches
265    }

328    fn ensure_transaction_open(&mut self) {
329        if self.transaction.is_none() {
330            let args = self.doc.transaction_args(self.isolation.as_deref());
331            let inner = TransactionInner::new(args);
332            self.transaction = Some((self.patch_log.branch(), inner))
333        }
334    }
335
336    fn ensure_transaction_closed(&mut self) {
337        if let Some((patch_log, tx)) = self.transaction.take() {
338            self.patch_log.merge(patch_log);
339            let hash = tx.commit(&mut self.doc, None, None);
340            if self.isolation.is_some() && hash.is_some() {
341                self.isolation = hash.map(|h| vec![h])
342            }
343        }
344    }

528    /// Commit the current operations with some options.
529    ///
530    /// Returns [`None`] if there were no operations to commit
531    ///
532    /// ```
533    /// # use automerge::transaction::CommitOptions;
534    /// # use automerge::transaction::Transactable;
535    /// # use automerge::ROOT;
536    /// # use automerge::AutoCommit;
537    /// # use automerge::ObjType;
538    /// # use std::time::SystemTime;
539    /// let mut doc = AutoCommit::new();
540    /// doc.put_object(&ROOT, "todos", ObjType::List).unwrap();
541    /// let now = SystemTime::now().duration_since(SystemTime::UNIX_EPOCH).unwrap().as_secs() as
542    /// i64;
543    /// doc.commit_with(CommitOptions::default().with_message("Create todos list").with_time(now));
544    /// ```
545    pub fn commit_with(&mut self, options: CommitOptions) -> Option<ChangeHash> {
546        // ensure that even no changes triggers a change
547        self.ensure_transaction_open();
548        let (patch_log, tx) = self.transaction.take().unwrap();
549        self.patch_log.merge(patch_log);
550        let hash = tx.commit(&mut self.doc, options.message, options.time);
551        if self.isolation.is_some() && hash.is_some() {
552            self.isolation = hash.map(|h| vec![h])
553        }
554        hash
555    }

599    fn get_scope(&self, heads: Option<&[ChangeHash]>) -> Option<Clock> {
600        // heads arg takes priority
601        if let Some(h) = heads {
602            return Some(self.doc.clock_at(h));
603        }
604        match (&self.isolation, &self.transaction) {
605            // then look at in progress isolated transaction
606            (Some(_), Some((_, t))) => t.get_scope().clone(),
607            // then look at clock for isolation
608            (Some(i), None) => Some(self.doc.clock_at(i)),
609            _ => None,
610        }
611    }

613    fn patch_to(&mut self, after: &[ChangeHash]) {
614        // we may be isolated so we dont use self.doc.get_heads()
615        let before = self.get_heads();
616        if before.as_slice() != after {
617            let before_clock = self.doc.clock_at(&before);
618            let after_clock = self.doc.clock_at(after);
619            diff::log_diff(&self.doc, &before_clock, &after_clock, &mut self.patch_log);
620        }
621    }

1034// A wrapper we return from [`AutoCommit::sync()`] to ensure that transactions are closed before we
1035// start syncing
1036struct SyncWrapper<'a> {
1037    inner: &'a mut AutoCommit,
1038}
1039
1040impl<'a> SyncDoc for SyncWrapper<'a> {
1041    fn generate_sync_message(&self, sync_state: &mut sync::State) -> Option<sync::Message> {
1042        self.inner.doc.generate_sync_message(sync_state)
1043    }
1044
1045    fn receive_sync_message(
1046        &mut self,
1047        sync_state: &mut sync::State,
1048        message: sync::Message,
1049    ) -> Result<(), AutomergeError> {
1050        self.inner.ensure_transaction_closed();
1051        if self.inner.isolation.is_some() {
1052            self.inner.doc.receive_sync_message_log_patches(
1053                sync_state,
1054                message,
1055                &mut PatchLog::null(),
1056            )
1057        } else {
1058            self.inner.doc.receive_sync_message_log_patches(
1059                sync_state,
1060                message,
1061                &mut self.inner.patch_log,
1062            )
1063        }
1064    }
1065
1066    // I dont like this function - it makes sense on automerge but not autocommit
1067    // FIXME
1068    fn receive_sync_message_log_patches(
1069        &mut self,
1070        sync_state: &mut sync::State,
1071        message: sync::Message,
1072        patch_log: &mut PatchLog,
1073    ) -> Result<(), AutomergeError> {
1074        self.inner
1075            .doc
1076            .receive_sync_message_log_patches(sync_state, message, patch_log)
1077    }
1078}

1080#[derive(Debug, Clone, PartialEq)]
1081struct OpRange {
1082    before_len: usize,
1083    hashes: Vec<ChangeHash>,
1084}
1085
1086impl OpRange {
1087    fn new(before: &[ChangeHash], after: &[ChangeHash]) -> Self {
1088        let mut hashes = Vec::with_capacity(before.len() + after.len());
1089        hashes.extend(before);
1090        hashes.extend(after);
1091        let range = Self {
1092            before_len: before.len(),
1093            hashes,
1094        };
1095        assert_eq!(before, range.before());
1096        assert_eq!(after, range.after());
1097        range
1098    }
1099
1100    fn before(&self) -> &[ChangeHash] {
1101        &self.hashes[0..self.before_len]
1102    }
1103
1104    fn after(&self) -> &[ChangeHash] {
1105        &self.hashes[self.before_len..]
1106    }
1107}
