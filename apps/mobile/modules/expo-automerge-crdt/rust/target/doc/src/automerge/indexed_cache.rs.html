6#[derive(Debug, Clone)]
7pub(crate) struct IndexedCache<T> {
8    pub(crate) cache: Vec<T>,
9    lookup: HashMap<T, usize>,
10}
11
12impl<T> PartialEq for IndexedCache<T>
13where
14    T: PartialEq,
15{
16    fn eq(&self, other: &Self) -> bool {
17        self.cache == other.cache
18    }
19}
20
21impl<T> IndexedCache<T>
22where
23    T: Clone + Eq + Hash + Ord,
24{
25    pub(crate) fn new() -> Self {
26        IndexedCache {
27            cache: Default::default(),
28            lookup: Default::default(),
29        }
30    }
31
32    pub(crate) fn cache(&mut self, item: T) -> usize {
33        if let Some(n) = self.lookup.get(&item) {
34            *n
35        } else {
36            let n = self.cache.len();
37            self.cache.push(item.clone());
38            self.lookup.insert(item, n);
39            n
40        }
41    }
42
43    pub(crate) fn lookup(&self, item: &T) -> Option<usize> {
44        self.lookup.get(item).cloned()
45    }
46
47    #[allow(dead_code)]
48    pub(crate) fn len(&self) -> usize {
49        self.cache.len()
50    }
51
52    pub(crate) fn get(&self, index: usize) -> &T {
53        &self.cache[index]
54    }
55
56    pub(crate) fn safe_get(&self, index: usize) -> Option<&T> {
57        self.cache.get(index)
58    }
59
60    #[allow(dead_code)]
61    pub(crate) fn sorted(&self) -> IndexedCache<T> {
62        let mut sorted = Self::new();
63        self.cache.iter().sorted().cloned().for_each(|item| {
64            let n = sorted.cache.len();
65            sorted.cache.push(item.clone());
66            sorted.lookup.insert(item, n);
67        });
68        sorted
69    }
70}

88impl<A: Hash + Eq + Clone> FromIterator<A> for IndexedCache<A> {
89    fn from_iter<T: IntoIterator<Item = A>>(iter: T) -> Self {
90        let mut cache = Vec::new();
91        let mut lookup = HashMap::new();
92        for (index, elem) in iter.into_iter().enumerate() {
93            cache.push(elem.clone());
94            lookup.insert(elem, index);
95        }
96        Self { cache, lookup }
97    }
