1use std::str::FromStr;
2
3use serde::{de, Deserialize, Deserializer, Serialize, Serializer};
4
5use crate::legacy::{ObjectId, OpId};
6
7impl Serialize for ObjectId {
8    fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
9    where
10        S: Serializer,
11    {
12        match self {
13            ObjectId::Id(id) => id.serialize(serializer),
14            ObjectId::Root => serializer.serialize_str("_root"),
15        }
16    }
17}
18
19impl<'de> Deserialize<'de> for ObjectId {
20    fn deserialize<D>(deserializer: D) -> Result<Self, D::Error>
21    where
22        D: Deserializer<'de>,
23    {
24        let s = String::deserialize(deserializer)?;
25        if s == "_root" {
26            Ok(ObjectId::Root)
27        } else if let Ok(id) = OpId::from_str(&s) {
28            Ok(ObjectId::Id(id))
29        } else {
30            Err(de::Error::invalid_value(
31                de::Unexpected::Str(&s),
32                &"A valid ObjectID",
33            ))
34        }
35    }
36}
