11#[derive(Debug, Clone, Default, PartialEq)]
12pub(crate) struct RichTextQueryState<'a> {
13    map: HashMap<OpId, &'a MarkData, FxBuildHasher>,
14    block: Option<OpId>,
15}
16
17impl<'a> RichTextQueryState<'a> {
18    pub(crate) fn process(&mut self, op: Op<'a>, clock: Option<&Clock>) {
19        if !(clock.map(|c| c.covers(op.id())).unwrap_or(true)) {
20            // if the op is not visible in the current clock
21            // we can ignore it
22            return;
23        }
24        match op.action() {
25            OpType::MarkBegin(_, data) => {
26                self.map.insert(*op.id(), data);
27            }
28            OpType::MarkEnd(_) => {
29                self.map.remove(&op.id().prev());
30            }
31            OpType::Make(ObjType::Map) => {
32                self.block = Some(*op.id());
33            }
34            _ => {}
35        }
36    }
37
38    pub(crate) fn iter(&self) -> impl Iterator<Item = (&OpId, &&MarkData)> {
39        self.map.iter()
40    }
41
42    pub(crate) fn insert(&mut self, op: OpId, data: &'a MarkData) {
43        self.map.insert(op, data);
44    }
45
46    pub(crate) fn remove(&mut self, op: &OpId) {
47        self.map.remove(op);
48    }
49}

51#[derive(Debug, Clone, PartialEq)]
52pub(crate) struct ListState {
53    encoding: ListEncoding,
54    last_seen: Option<Key>,
55    last_width: usize,
56    never_seen_puts: bool,
57    target: usize,
58    index: usize,
59    pos: usize,
60}
61
62// There are two indexes being tracked in lists
63// pos: this is the position in the opset.  A list of 100 items may have 1000 ops.  Each op has a position
64// index: this is the logical index of the list of visible values.  Conflicted items will have the same index.
65//        The index is affected by utf8/utf16 encoding for text, etc
66
67impl ListState {
68    pub(crate) fn new(encoding: ListEncoding, target: usize) -> Self {
69        ListState {
70            encoding,
71            target,
72            last_seen: None,
73            last_width: 0,
74            index: 0,
75            pos: 0,
76            never_seen_puts: true,
77        }
78    }
79
80    pub(crate) fn was_last_seen(&self, key: Key) -> bool {
81        self.last_seen == Some(key)
82    }
83
84    // lists that have never seen puts (only inserts and deletes)
85    // can take advantage of a faster codepath
86    pub(crate) fn check_if_node_is_clean(&mut self, index: &Index) {
87        self.never_seen_puts &= index.has_never_seen_puts();
88    }
89
90    pub(crate) fn process_node<'a>(
91        &mut self,
92        node: &'a OpTreeNode,
93        index: &'a Index,
94        osd: &OpSetData,
95        marks: Option<&mut RichTextQueryState<'a>>,
96    ) -> QueryResult {
97        if self.encoding == ListEncoding::List {
98            self.process_list_node(node, index, osd, marks)
99        } else if self.never_seen_puts {
100            // text node is clean - use the indexes
101            self.process_text_node(node, index, marks)
102        } else {
103            // text nodes are intended to only be interacted with splice()
104            // meaning all ops are inserts or deleted inserts
105            // the indexes are written with this assumption in mind
106            // if conflicted put()'s with different character widths exist
107            // we cannot trust the indexs and need to descend
108            QueryResult::Descend
109        }
110    }
111
112    fn process_marks<'a>(&mut self, index: &'a Index, marks: Option<&mut RichTextQueryState<'a>>) {
113        if let Some(marks) = marks {
114            for (id, data) in index.mark_begin.iter() {
115                marks.insert(*id, data);
116            }
117            for id in index.mark_end.iter() {
118                marks.remove(&id.prev());
119            }
120        }
121    }
122
123    fn process_text_node<'a>(
124        &mut self,
125        node: &'a OpTreeNode,
126        index: &'a Index,
127        marks: Option<&mut RichTextQueryState<'a>>,
128    ) -> QueryResult {
129        let num_vis = index.visible_len(self.encoding);
130        if self.index + num_vis >= self.target {
131            return QueryResult::Descend;
132        }
133        self.index += num_vis;
134        self.pos += node.len();
135        self.process_marks(index, marks);
136        QueryResult::Next
137    }
138
139    fn process_list_node<'a>(
140        &mut self,
141        node: &'a OpTreeNode,
142        index: &'a Index,
143        osd: &OpSetData,
144        marks: Option<&mut RichTextQueryState<'a>>,
145    ) -> QueryResult {
146        let mut num_vis = index.visible.len();
147        if let Some(last_seen) = self.last_seen {
148            // the elemid `last_seen` is counted in this node's index
149            // but since we've already seen it we dont want to count it again
150            // as this is only for lists we can subtract 1
151            if index.has_visible(&last_seen) {
152                num_vis -= 1;
153            }
154        }
155
156        if self.index + num_vis >= self.target {
157            // if we've reached out target - decend
158            return QueryResult::Descend;
159        }
160
161        self.index += num_vis;
162        self.pos += node.len();
163
164        // scenario 1: the last elemid is in the index - record it
165        // scenario 2: the last elemid this node is not in the index
166        // scenario 2a: and its different than the previous elemid - last_seen = None
167        //              as there is no possability of visible elements spanning the
168        //              node boundary
169        // scenario 2b: it is the same as the previous last_seen - do nothing
170
171        let last_elemid = node.last().as_op(osd).elemid_or_key();
172        if index.has_visible(&last_elemid) {
173            self.last_seen = Some(last_elemid);
174        } else if self.last_seen.is_some() && Some(last_elemid) != self.last_seen {
175            self.last_seen = None;
176        }
177        self.process_marks(index, marks);
178        QueryResult::Next
179    }
180
181    pub(crate) fn process_op(&mut self, op: Op<'_>, current: Key, visible: bool) {
182        if visible {
183            if self.never_seen_puts {
184                // clean sequnces are simple - only insert and deletes
185                self.last_width = op.width(self.encoding);
186                self.index += self.last_width;
187            } else {
188                let current_width = op.width(self.encoding);
189                if self.last_seen != Some(current) {
190                    // new value - progess
191                    self.last_width = current_width;
192                    self.index += self.last_width;
193                    self.last_seen = Some(current);
194                } else if current_width != self.last_width {
195                    // width is always 1 for lists so this
196                    // will only trigger if there are conflicting unicode characters
197                    // of different lengths
198                    self.index = self.index + current_width - self.last_width;
199                    self.last_width = current_width;
200                }
201            }
202        }
203        self.pos += 1;
204    }
205
206    pub(crate) fn target(&self) -> usize {
207        self.target
208    }
209
210    pub(crate) fn pos(&self) -> usize {
211        self.pos
212    }
213
214    pub(crate) fn index(&self) -> usize {
215        self.index
216    }
217
218    pub(crate) fn last_index(&self) -> usize {
219        self.index - self.last_width
220    }
221
222    pub(crate) fn done(&self) -> bool {
223        self.index >= self.target
224    }
225
226    pub(crate) fn seek(&mut self, last: &LastInsert) {
227        self.last_width = last.width;
228        self.index = last.index + last.width;
229        self.pos = last.pos + 1;
230    }
