1use std::convert::Infallible;
2
3/// Represents an error which occurred when splicing.
4///
5/// When splicing values into existing column storage there are two kinds of errors which can
6/// occur, those caused by iterating over the existing items, and those caused by iterating over
7/// the replacement items.
8#[derive(Debug)]
9pub(crate) enum SpliceError<E, R> {
10    /// There was an error reading from the existing column storage
11    ReadExisting(E),
12    /// There was an error reading from the iterator of new rows
13    ReadReplace(R),
14}
15
16impl<E> SpliceError<E, Infallible> {
17    /// Map a spliceerror which is infallible in it's `Replace` error type into a different error.
18    ///
19    /// This is used when you have performed a splice with a `replace` iterator which is
20    /// infallible and need to return a more general `SpliceError`
21    pub(crate) fn existing<R>(self) -> SpliceError<E, R> {
22        match self {
23            SpliceError::ReadExisting(e) => SpliceError::ReadExisting(e),
24            SpliceError::ReadReplace(_) => unreachable!("absurd"),
25        }
26    }
27}
28
29impl<E, R> std::error::Error for SpliceError<E, R>
30where
31    E: std::error::Error,
32    R: std::error::Error,
33{
34}
35
36impl<E, R> std::fmt::Display for SpliceError<E, R>
37where
38    E: std::fmt::Display,
39    R: std::fmt::Display,
40{
41    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
42        match self {
43            Self::ReadExisting(e) => write!(f, "error reading from existing rows: {}", e),
44            Self::ReadReplace(e) => write!(f, "error reading from replacement rows: {}", e),
45        }
46    }
47}
